  1. Create a new empty web page 0.html.
  2. Paste the code below into the web page.
  3. Execute the web page with the Google Chrome debugger. You need to be able to describe what the javascript code is doing!
  4. Bonus! Change the Javascript code so it prints the message in the <pre> element in the web page, and not in a message box! 
                        <html>
                            <head>
                                <script>
                                    var myNumber = 125;

                                    function three_a_plus_one (a)
                                    {
                                        var a0;
                                        var div2;
                                        var nRuns;
                                        var msg;

                                        a0      = a;
                                        nRuns   = 0;
                                        while (a0 >= 2)
                                        {
                                            div2 = a0  %  2;
                                            if (div2 === 0)
                                            {
                                                a0 = a0 / 2;
                                            }
                                            else
                                            {
                                                a0 = 3 * a0 + 1;
                                            }
                                            nRuns++;
                                        }

                                        msg = "For number " + a + " we needed " + nRuns + " iterations.";
                                        window.alert (msg);
                                    }
                                </script>
                            </head>
                            <body onload="three_a_plus_one (myNumber);">
                                <pre></pre>
                            </body>
                        </html>
