1 /** 
  2  *  @fileoverview        Storage, holding all {@link TExercise} objects in a worksheet.
  3  */
  4 define 
  5 (
  6     [
  7         "dojo/_base/declare"
  8     ],
  9     function 
 10     (
 11         declare
 12     )
 13     {
 14         /**
 15          * Type of test to do when we check whether any given UID is already in storage.
 16          * 
 17          * @enum
 18          */
 19         var ETestType =
 20         {
 21             kSkip:              0,      /* Don't test                               */
 22             kIDMustExist:       1,      /* If UID does NOT exist, throw exception   */
 23             kIDMustNotExist:    2       /* If UID DOES exist, throw exception       */
 24         };
 25         
 26         var TModel;
 27         var ret;
 28 
 29         /**
 30          * Storage model, holding all exercises of a worksheet. Provides index 
 31          * based access and key based access.
 32          * 
 33          * @class       TModel
 34          */
 35         TModel = 
 36         {
 37             /**
 38              * Background storage for index based access.
 39              * 
 40              * @type        JSArray
 41              * @private
 42              */
 43             fExerciseList:          null,
 44 
 45             /**
 46              * Background storage for key based access.
 47              * 
 48              * @type        JSObject
 49              * @private
 50              */
 51             fExerciseMap:           null,
 52             
 53             /**
 54              * Returns the exercise with the given key, <code>id</code>.
 55              * 
 56              * @param   {String}        id      The UID of the desired exercise.
 57              * @returns {TExtercise}    The exercise.
 58              * @thows   (String)        exception with error message if there
 59              *                          isn't any such exercise.
 60              */
 61             GetByID: function (id)
 62             {
 63                 var ret;
 64                 
 65                 this._AssertIDUsable(id, ETestType.kIDMustExist);
 66                 ret = this.fExerciseMap [id];
 67                 
 68                 return ret;
 69             },
 70             
 71             /**
 72              * Returns the exercise with the given index, <code>i</code>. Index 
 73              * must be between <code>0</code> (zero) and <code>n-1</code>, where
 74              * <code>n</code> is the amount of exercises stored.
 75              * 
 76              * @param   {String}        i       The index of the desired exercise.
 77              * @returns {TExtercise}    The exercise.
 78              * @thows   (String)        exception with error message if the given 
 79              *                          index is out of bounds.
 80              */
 81             GetByIndex: function (i)
 82             {
 83                 var ret;
 84                 
 85                 this._AssertIndexUsable (i);
 86                 ret = this.fExerciseList [i];
 87                 
 88                 return ret;
 89             },
 90             
 91             /**
 92              * Returns the number of exercises stored.
 93              * 
 94              * @returns {integer}       The number of exercises.
 95              */
 96             GetNumElements: function ()
 97             {
 98                 return this.fExerciseList.length;
 99             },
100             
101             /**
102              * Returns <code>true</code>, if there's an exercise with the 
103              * given <code>id</code>, <code>false</code> otherwise.
104              * 
105              * @param       {String}    id      The UID of the exercise searched for.
106              * @returns     {TExercise}         The exercise searched, or, 
107              *                                  <code>null</code> if no such exercise 
108              *                                  is stored.
109              */
110             HasElement: function (id)
111             {
112                 var ret;
113                 
114                 ret = this.fExerciseMap.hasOwnProperty (id);
115                 
116                 return ret;
117             },
118             
119             /**
120              * Registers the given exercise, i.e. adds it to storage.
121              * 
122              * @param   {TExercise}     exercise    The exercise to be stored.
123              * @throws  {String}        Exception with message if an exercise with
124              *                          the same UID is already stored.
125              */
126             Register: function (exercise)
127             {
128                 this._AssertIDUsable (exercise.fID, ETestType.kIDMustNotExist);
129                 exercise.fIsNullObject = false;
130                 this.fExerciseList.push (exercise);
131                 this.fExerciseMap [exercise.fID] = exercise;
132             },
133             
134             /**
135              * cTor.
136              */
137             constructor: function ()
138             {
139                 this.fExerciseList  = [];
140                 this.fExerciseMap   = {}; 
141             },
142 
143             /**
144              * Tests whether the given <code>id</code> is aready stored, resp. not stored. 
145              * The exact behaviour depends on the <code>testTypeUnique</code> parameter: 
146              * 
147              * <table border="1">
148              *     <tr>
149              *         <th>testTypeUnique</th>
150              *         <th>hasID</th>
151              *         <th>Result</th>
152              *     </tr>
153              *     <tr>
154              *         <td><code>ETestType.kSkip</code></td>
155              *         <td><code>true</code></td>
156              *         <td>OK</td>
157              *     </tr>
158              *     <tr>
159              *         <td> </td>
160              *         <td><code>false</code></td>
161              *         <td>OK</td>
162              *     </tr>
163              *     <tr>
164              *         <td><code>ETestType.kIDMustExist</code></td>
165              *         <td><code>true</code></td>
166              *         <td>OK</td>
167              *     </tr>
168              *     <tr>
169              *         <td> </td>
170              *         <td><code>false</code></td>
171              *         <td>throw exception</td>
172              *     </tr>
173              *     <tr>
174              *         <td><code>ETestType.kIDMustNotExist</code></td>
175              *         <td><code>true</code></td>
176              *         <td>throw exception</td>
177              *     </tr>
178              *     <tr>
179              *         <td> </td>
180              *         <td><code>false</code></td>
181              *         <td>OK</td>
182              *     </tr>
183              * </table>
184              * 
185              * @param   {type}        id                  The UID to look for.
186              * @param   {type}        testTypeUnique      A flag, determining the test specifics.
187              * @private
188              * @throws {String}       Exception with message if an exercise with the same 
189              *                        ID does (not) exist.
190              */
191             _AssertIDUsable: function (id, testTypeUnique)
192             {
193                 var kID = "TModel::_AssertIDUsable";
194                 
195                 var isWrongType;
196                 var isNull;
197                 var hasID;
198 
199                 isWrongType = (typeof id !== 'string');
200                 isNull      = (id == null);
201                 if (isWrongType)
202                 {
203                     throw kID + ": Exercise ID must be a string. Given: " + (typeof id);
204                 }
205                 else if (isNull)
206                 {
207                     throw kID + ": Exercise ID must be not null. Given: null.";
208                 }
209 
210                 /**
211                  * Test whether id already exists. Behaviour:
212                  * 
213                  * testType                     hasID       Result
214                  * ---------------------------------------------------------
215                  *  ETestType.kSkip             true        OK
216                  *                              false       OK
217                  *  ETestType.kIDMustExist      true        OK
218                  *                              false       throw exception
219                  *  ETestType.kIDMustNotExist   true        throw exception
220                  *                              false       OK
221                  */
222                 if (testTypeUnique !== ETestType.kSkip)
223                 {
224                     hasID = this.fExerciseMap.hasOwnProperty (id);
225                     if (testTypeUnique === ETestType.kIDMustExist)
226                     {
227                         if (! hasID)
228                         {
229                             throw kID + ": Missing exercise ID: " + id;
230                         }
231                     }
232                     else
233                     {
234                         if (hasID)
235                         {
236                             throw kID + ": Exercise ID must be unique. Offending ID: " + id;
237                         }
238                     }
239                 }
240             },
241             
242             /**
243              * Tests whether the given index <code>i</code> is within bounds.
244              * For <code>i</code> to be valid, the value of <code>i</code> must
245              * be in interval <code>[0, n[</code> where <code>n</code>
246              * is the number of exercises stored.
247              * 
248              * @param   {type}        i                   The index to test.
249              * @private
250              * @throws {String}       Exception with message if <code>i</code>
251              *                        is out of bounds.
252              */
253             _AssertIndexUsable: function (i)
254             {
255                 var kID = "TModel::_AssertIndexUsable";
256                 
257                 var isWrongType;
258                 var n;
259                 
260                 isWrongType = ((typeof i !== "number") || ((i % 1) != 0));
261                 if (isWrongType)
262                 {
263                     throw kID + ": Given index has wrong type. Requirement: Must be an integer. i: " + i + "[" + (typeof i) + "]";
264                 }
265                 
266                 n = this.fExerciseList.length;
267                 if (i < 0)
268                 {
269                     throw kID + ": Index too small. Requirement: 0 <= i < nExercises. i: " + i + ", nExercises = " + n;
270                 }
271                 else if (i >= n)
272                 {
273                     throw kID + ": Index too large. Requirement: 0 <= i < nExercises. i: " + i + ", nExercises = " + n;
274                 }
275             }
276         };
277     
278         ret = declare ("TModel", [], TModel);
279     
280         return ret;
281     }
282 );
