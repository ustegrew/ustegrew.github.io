372             /**
373              * For the given event, executes state transition. Transition is executed in several steps:
374              * 
375              * 1. Worksheet executes pre transition actions (Save current exercise, collapse current solution etc.).
376              * 2. Worksheet sends "kSuccess" notification.
377              * 3. Now perform the transition. This also calls any worksheet specific handler.
378              */
379             Notify: function (event)
380             {
381                 var t
382                 
383                 if (gDebug)                                            /* [20] */
384                     {console.log ("TController::Notify ('" + event + "')");}
385 
386                 /* It can be confusing, but for each transition the 'else' branch executes first.
387                  * Once all the pre transition actions have completed, the worksheet will send
388                  * a 'kSuccess' notification which will be handled in the TRUE branch of the 
389                  * if statement. */
390                 if (event === "kSuccess")
391                 {
392                     /* Worksheet has finished animation actions. */
393                     /* We get our schedule transition back from the stack */
394                     t = this.fStack.pop ();
395                     
396                     if (gDebug)                                        /* [20] */
397                         {console.log ("TController::Notify ('" + event + "'): " + this.fState + " [" + event + "] -> " + t.stateNext);}
398 
399                     /* Now execute transition */
400                     this.fState = t.stateNext;
401                     t.handler.call (this.fHost);                            /* [10] */
402                 }
403                 else
404                 {
405                     /* Schedule transition on stack. Will be handled after animations have completed. */
406                     t = this._GetTransition (event);
407                     this.fStack.push (t);
408                     
409                     if (gDebug)                                        /* [20] */
410                         {console.log ("TController::Notify ('" + event + "'): Executing pre transition actions");}
411 
412                     /* Now execute all animation actions. When they have completed, 
413                      * we will receive a "kSuccess" notification */
414                     this.fHost.NotifyPreTransitionActions (t.actions);                    
415                 }
416             },
417             
418             /**
419              * Returns the transition to be executed next.
420              * 
421              * @param   {type}      event
422              * @returns {JSON}      The transition to execute.
423              */
424             _GetTransition: function (event)
425             {
426                 var params;
427                 var stP;
428                 var st;
429                 var iNSt;
430                 var nNSt;
431                 var trans;
432                 var foundEvent;
433                 var ret;
434                 
435                 params              = (typeof args != 'undefined') ? args : null;
436                 st                  = gkStateTable [this.fState];
437                 nNSt                = st.transitions.length;
438                 foundEvent          = false;
439                 ret                 = null;
440                 if (nNSt >= 1)
441                 {
442                     for (iNSt = 0; ((iNSt < nNSt) && (! foundEvent)); iNSt++)
443                     {
444                         trans = st.transitions [iNSt];
445                         if (! foundEvent)
446                         {
447                             if (trans.event == event)
448                             {
449                                 foundEvent  = true;
450                                 ret         = trans;
451                             }
452                         }
453                     }
454                 }
455 
456                 if (foundEvent)
457                 {
458                     if (gDebug)
459                     {
460                         console.log ("TController::_GetTransition ('" + event + "')");
461                         console.log ("Found: " + JSON.stringify (ret));
462                     };
463                 }
464                 else
465                 {
466                     console.log 
467                     (
468                         "TController::_GetTransition ('" + event + "'): Invalid transition: " + this.fState + " [" + event + "]"
469                     );
470                 }
471                 
472                 return ret;
473             },

488                 this.fState = "kNull";
489                 this.fStack = [];
490             },
