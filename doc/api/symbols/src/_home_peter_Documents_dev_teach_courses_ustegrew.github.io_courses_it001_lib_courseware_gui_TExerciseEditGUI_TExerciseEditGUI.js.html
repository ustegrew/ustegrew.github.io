 40          * The editor's content change observer. Observes the associated editor for
 41          * changes of content. 
 42          * 
 43          * @class           TObserver_ContentChanged
 44          * @private
 45          * @see [110]
 46          */
 47         var TObserver_ContentChanged = function (host)                          
 48         {
 49             /* Local reference to 'this' - for callbacks, 
 50              * so we can run them in the proper context 
 51              */
 52             var _this               = this;
 53             
 54             /**
 55              * Interval [ms] between main loop cycles.
 56              */
 57             var kObserverInterval   = 1000;                                          /* [110] */
 58             
 59             /**
 60              * Possible states of this observer.
 61              * 
 62              * @private
 63              */
 64             var EObserverState =
 65             {
 66                 kWait:          10,     /* Observer inactive, i.e. not observing.   */
 67                 kRunning:       20      /* Observer active, i.e. observing.         */
 68             }
 69             
 70             /**
 71              * Content from previous observation.
 72              * 
 73              * @type    String
 74              * @private
 75              */
 76             this.fContentOld    = null;
 77             
 78             /**
 79              * The client using this observer
 80              * 
 81              * @type courseware/gui/TExerciseEditGUI
 82              * @private
 83              */
 84             this.fHost          = host;
 85             
 86             /**
 87              * The current state of this observer.
 88              * 
 89              * @type    EObserverState
 90              * @private
 91              */
 92             this.fState         = EObserverState.kWait; 
 93             
 94             /**
 95              * Sets this observer's state to inactive.
 96              */
 97             this.SetPaused = function ()
 98             {
 99                 if (gDebug) console.log ("TExerciseEditGUI::TObserver_ContentChanged::SetPaused ()");
100 
101                 _this.fState = EObserverState.kWait;
102             };
103             
104             /**
105              * Sets this observer's state to active.
106              */
107             this.SetRunning = function ()
108             {
109                 if (gDebug)
110                 {
111                     if (_this.fState != EObserverState.kRunning)
112                     {
113                         console.log ("TExerciseEditGUI::TObserver_ContentChanged::SetRunning ()");
114                     }
115                     else
116                     {
117                         console.log ("TExerciseEditGUI::TObserver_ContentChanged::SetRunning (): Cancelled as we are already in run state.");
118                     }
119                 }
120 
121                 if (_this.fState != EObserverState.kRunning)                    /* [112] */
122                 {
123                     _this.fContentOld   = this.fHost.fAPIEditor.GetContent ();
124                     _this.fState        = EObserverState.kRunning;
125                     _this._Run ();
126                 }
127             }
128             
129             /**
130              * The central loop.  Queries the editor's content and fires an onChange event if
131              * the editor's content has changed.
132              */
133             this._Run = function ()
134             {
135                 var content;
136                 var doQuery;
137                 
138                 doQuery     = this.fState == EObserverState.kRunning    &&
139                               this.fHost.fAPIEditor.fHasEditor; //      && what else ?
140                 
141                 if (doQuery)
142                 {
143                     content = this.fHost.fAPIEditor.GetContent ();
144                     if (content !== this.fContentOld)                           /* [111] */
145                     {
146                         this.fContentOld                   = content;
147                         this.fHost.fAPIEditor.fHasChanged  = true;
148                         this.fHost.fHandlers.onFinishedChange.call (this.fHost.fHost);
149                     }
150                     window.setTimeout (function () {_this._Run.call (_this)}, kObserverInterval);
151                 }
152             };
153             
154             /* We begin this observer in inactive state. */
155             this.fState = EObserverState.kWait;
156         };
