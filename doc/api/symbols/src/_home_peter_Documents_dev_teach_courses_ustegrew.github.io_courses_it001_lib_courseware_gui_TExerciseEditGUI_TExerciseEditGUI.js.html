 41         /**
 42          * The editor's content change observer. Observes the associated editor for
 43          * changes of content. 
 44          * 
 45          * @class           TObserver_ContentChanged
 46          * @private
 47          * @see [110]
 48          */
 49         var TObserver_ContentChanged = function (host)                          
 50         {
 51             /* Local reference to 'this' - for callbacks, 
 52              * so we can run them in the proper context 
 53              */
 54             var _this               = this;
 55             
 56             /**
 57              * Interval [ms] between main loop cycles.
 58              */
 59             var kObserverInterval   = 1000;                                          /* [110] */
 60             
 61             /**
 62              * Possible states of this observer.
 63              * 
 64              * @enum        {int}
 65              * @private
 66              */
 67             var EObserverState =
 68             {
 69                 kWait:          10,     /* Observer inactive, i.e. not observing.   */
 70                 kRunning:       20      /* Observer active, i.e. observing.         */
 71             }
 72             
 73             /**
 74              * Content from previous observation.
 75              * 
 76              * @type    String
 77              * @private
 78              */
 79             this.fContentOld    = null;
 80             
 81             /**
 82              * The client using this observer
 83              * 
 84              * @type courseware/gui/TExerciseEditGUI
 85              * @private
 86              */
 87             this.fHost          = host;
 88             
 89             /**
 90              * The current state of this observer.
 91              * 
 92              * @type    EObserverState
 93              * @private
 94              */
 95             this.fState         = EObserverState.kWait; 
 96             
 97             /**
 98              * Sets this observer's state to inactive.
 99              */
100             this.SetPaused = function ()
101             {
102                 if (gDebug) console.log ("TExerciseEditGUI::TObserver_ContentChanged::SetPaused ()");
103 
104                 _this.fState = EObserverState.kWait;
105             };
106             
107             /**
108              * Sets this observer's state to active.
109              */
110             this.SetRunning = function ()
111             {
112                 if (gDebug)
113                 {
114                     if (_this.fState != EObserverState.kRunning)
115                     {
116                         console.log ("TExerciseEditGUI::TObserver_ContentChanged::SetRunning ()");
117                     }
118                     else
119                     {
120                         console.log ("TExerciseEditGUI::TObserver_ContentChanged::SetRunning (): Cancelled as we are already in run state.");
121                     }
122                 }
123 
124                 if (_this.fState != EObserverState.kRunning)                    /* [112] */
125                 {
126                     _this.fContentOld   = this.fHost.fAPIEditor.GetContent ();
127                     _this.fState        = EObserverState.kRunning;
128                     _this._Run ();
129                 }
130             }
131             
132             /**
133              * The central loop.  Queries the editor's content and fires an onChange event if
134              * the editor's content has changed.
135              */
136             this._Run = function ()
137             {
138                 var content;
139                 var doQuery;
140                 
141                 doQuery     = this.fState == EObserverState.kRunning    &&
142                               this.fHost.fAPIEditor.fHasEditor; //      && what else ?
143                 
144                 if (doQuery)
145                 {
146                     content = this.fHost.fAPIEditor.GetContent ();
147                     if (content !== this.fContentOld)                           /* [111] */
148                     {
149                         this.fContentOld                   = content;
150                         this.fHost.fAPIEditor.fHasChanged  = true;
151                         this.fHost.fHandlers.onFinishedChange.call (this.fHost.fHost);
152                     }
153                     window.setTimeout (function () {_this._Run.call (_this)}, kObserverInterval);
154                 }
155             };
156             
157             /* We begin this observer in inactive state. */
158             this.fState = EObserverState.kWait;
159         };

167          * show either the ACE editor component or a Rich text editor.

180          *     <li>

182          *         editor with basic syntax checking for HTML code.

184          *     <li>
185          *         src/plain_text: Plain text. UI will show a source code  
186          *         editor without any syntax checking.
187          *     </li>
188          * </ul>
189          * <p><br/></p>
190          * 
191          * <b>Elements on the user interface</b>
192          * <pre>    
193          *     .------.--------------------------------------.
194          *     | File |                                      |
195          *     .---------------------------------------------.
196          *     | Content panel                               |
197          *     |                                             |
198          *     |                                             |
199          *     |                                             |
200          *     |                                             |
201          *     |                                             |
202          *     |                                             |
203          *     |                                             |
204          *     |                                             |
205          *     '---------------------------------------------'
206          * </pre>
207          * 
208          * <dl>
209          *     <dt>Menu bar</dt>
210          *     <dd>For actions such as "Save", "Close", ...</dd>
211          *     
212          *     <dt>Content panel</dt>
213          *     <dd>The content of the user's solution.</dd>
214          * </dl>
215          * <p><br/></p>
216          *

218          * the environment. For example, if we run on anything but Chrome and Firefox

222          * <table border="1">
223          *     <tr>
224          *         <th>Browser used</th>
225          *         <th>Specifics</th>
226          *     </tr>
227          *     <tr>
228          *         <td>Chrome or Firefox</td>
229          *         <td>All features offered.</td>
230          *     </tr>
231          *     <tr>
232          *         <td>Anything but Chrome or Firefox</td>
233          *         <td>Rich text editor has no formatting capabilities.</td>
234          *     </tr>
235          * </table>
236          * <p><br/></p>
237          * 
238          * <b>Example use - Instantiate editor, set type, set content, get content.</b>
239          * @example

253          *         var cnt;

255          *         // Create editor instance
256          *         editor = new TExerciseEditGUI
257          *         (
258          *             {
259          *                 fHost:              window,
260          *                 fWidth:             "745px"
261          *                 fHeight:            "480px"
262          *                 onFinishedChange:   function () {console.log ("Content changed.");},
263          *                 onFinishedLoad:     function () {console.log ("Text loaded.");},
264          *                 onRequestCancel:    function () {console.log ("User requested to close the document.");},
265          *                 onRequestSave:      function () {console.log ("User requested to save the document.");}
266          *             }
267          *         );
268          *         editor.startup ();
269          * 
270          *         // Embed editor in DOM element on hosting page. Assume, element has ID "pnlEditor"
271          *         domConstruct.place (editor.domNode, "pnlEditor", "only");
272          * 
273          *         // Set editor to be a javascript source code editor.  Set content.
274          *         editor.SetType      ("src", "js");

282          *         cnt = editor.GetContent ();
283          *         console.log (cnt);
284          * 
285          *

292          *

295          *
