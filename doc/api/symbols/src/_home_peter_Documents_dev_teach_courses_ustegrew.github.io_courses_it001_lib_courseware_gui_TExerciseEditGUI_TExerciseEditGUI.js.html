 49          * The editor's content change observer. Observes the current editor for
 50          * changes of content. 
 51          * 
 52          * For more details: [110]
 53          */
 54         var TObserver_ContentChanged = function (host)                          
 55         {
 56             var _this = this;
 57             
 58             var EObserverState =
 59             {
 60                 kWait:          10,
 61                 kRunning:       20
 62             }
 63             
 64             this.fContentOld    = null;
 65             this.fHost          = host;
 66             this.fState         = EObserverState.kWait; 
 67             
 68             this.SetPaused = function ()
 69             {
 70                 if (gDebug) console.log ("TExerciseEditGUI::TObserver_ContentChanged::SetPaused ()");
 71 
 72                 _this.fState = EObserverState.kWait;
 73             };
 74             
 75             this.SetRunning = function ()
 76             {
 77                 if (gDebug)
 78                 {
 79                     if (_this.fState != EObserverState.kRunning)
 80                     {
 81                         console.log ("TExerciseEditGUI::TObserver_ContentChanged::SetRunning ()");
 82                     }
 83                     else
 84                     {
 85                         console.log ("TExerciseEditGUI::TObserver_ContentChanged::SetRunning (): Cancelled as we are already in run state.");
 86                     }
 87                 }
 88 
 89                 if (_this.fState != EObserverState.kRunning)                    /* [112] */
 90                 {
 91                     _this.fContentOld   = this.fHost.fAPIEditor.GetContent ();
 92                     _this.fState        = EObserverState.kRunning;
 93                     _this._Run ();
 94                 }
 95             }
 96             
 97             this._Run = function ()
 98             {
 99                 var content;
100                 var doQuery;
101                 
102                 doQuery     = this.fState == EObserverState.kRunning    &&
103                               this.fHost.fAPIEditor.fHasEditor; //      && what else ?
104                 
105                 if (doQuery)
106                 {
107                     content = this.fHost.fAPIEditor.GetContent ();
108                     if (content !== this.fContentOld)                           /* [111] */
109                     {
110                         this.fContentOld                   = content;
111                         this.fHost.fAPIEditor.fHasChanged  = true;
112                         this.fHost.fHandlers.onFinishedChange.call (this.fHost.fHost);
113                     }
114                     window.setTimeout (function () {_this._Run.call (_this)}, kObserverInterval);
115                 }
116             };
117             
118             this.fState = EObserverState.kWait;
119         };

729 /*
730 
731 [10]:   All event handlers run in the context of the 'host' object provided by the 
732         client that created this object when doing cTor call such as: 
733             myExercise = new TExerciseGUI ({fHost:someObject}
734                         
735 [20]:   For editor detection we use a flag that is set to false at the earliest 
736         possible moment (i.e. after constructor call or immediately after destroying
737         any existing editor) and set to true as late as possible (i.e. after the
738         asynchronous loading of an editor component is complete. This way we ensure
739         reliable editor detection. Testing on this.fAPIEditor.fEditor == null is not
740         enough, because in
741         
742             this.fAPIEditor.fEditor = new TEditor ()
743         
744         this.fAPIEditor.fEditor may be set to something even if the
745         concrete editor hasn't completed loading yet.
746         Due to the asynchronous loading we may get problems when trying to access
747         the actual editor component too early (e.g. Null exception when trying to 
748         get or set the content of the editor even though that one hasn't finished
749         loading yet).
750                         
751 [21]:   Clients might try to set the content whilst the actual editor component 
752         hasn't finished loading. In that case we cache the content-to-be in a
753         property and set the editor content inside the resp. editor's onLoad 
754         handler.
755                         
756 [22]:   Bad solution. Forces all clients to specify these values (either hard code 
757         or implement some discovery mechanism). Better to let this component discover
758         the value automatically.
759         Logged as issue#12 on github [https://github.com/ustegrew/ustegrew.github.io/issues/12]
760         
761 [23]:   We won't log the call to the Destroy function, even if _SetType() hasn't been 
762         called yet. Otherwise, calling _SetType () for the first time would create a
763         log entry (which is annoying). This could be mitigated by introducing some flag 
764         (e.g. boolean fHasBeenSetBefore). However, the benefits of this feature are 
765         negligible and the cost is yet another obscure feature (makes the code harder to 
766         maintain). Not worth doing.        
767 
768 [100]:  It's important to set the editor's event handlers BEFORE creating the 
769         actual editor. This is critical, especially with the Ace editor, as that 
770         one loads asynchronously and will invoke the onLoad handler at the wrong

807 [110]:  We cannot use the onChange event as provided by the dijit/Editor component - that
808         one only fires if the content has changed AND the component looses focus. In our
809         case we need to fire the onChange event as soon as something changed, even if the
