1 /**
  2  *  @fileoverview        Data storage driver. This driver passes all data unmodified.
  3  */
  4 define 
  5 (
  6     [
  7         "dojo/_base/declare",
  8         "pieroxy_lz-string/lz-string",
  9         "./TDriver"
 10     ],
 11     function 
 12     (
 13         declare,
 14         compressor,
 15         TDriver
 16     )
 17     {
 18         var kEconomyMin = 20;
 19         
 20         var TDriverCompress;
 21         var ret;
 22 
 23         /**
 24          * The compression driver. (De)Compresses all data to (from) the local 
 25          * storage. 
 26          * Not all data will be compressed! For each submitted chunk of data 
 27          * the driver decides whether to store it compressed or as-is. This is
 28          * decided by the amount of storage saved: For each submitted chunk:
 29          *     - if compressing can reduce storage size by at least 20% then 
 30          *       the data will be stored in compressed format.
 31          *     - if savings are less than 20%, then we store uncompressed.
 32          * After the compression round the data will be packed into a JS object:
 33          *     {
 34          *         isCompressed:    boolean     // true if data is compressed
 35          *         data:            string      // The data payload
 36          *     }
 37          * This object will be (de)serialized to/from a JSON record.
 38          * 
 39          * @class       TDriverCompress
 40          */
 41         TDriverCompress = 
 42         {
 43             Get: function (key)
 44             {
 45                 var stor;
 46                 var record;
 47                 var ret;
 48                 
 49                 ret  = null;
 50                 stor = this.fStore.getItem (key);
 51                 if (stor !== null)
 52                 {
 53                     record  = JSON.parse (stor);
 54                     if (record.isCompressed)
 55                     {
 56                         ret = compressor.decompressFromBase64 (record.data);
 57                     }
 58                     else
 59                     {
 60                         ret = record.data;
 61                     }
 62                 }
 63                 
 64                 return ret;
 65             },
 66         
 67             Set: function (key, data)
 68             {
 69                 var compr;
 70                 var lPerc;
 71                 var economy;
 72                 var record;
 73                 var stor;
 74                 
 75                 compr   = compressor.compressToBase64 (data);                   /* [10] */
 76                 if (compr.length >= 1)
 77                 {
 78                     lPerc   = 100 * compr.length / data.length;
 79                     economy = 100 - lPerc;
 80                     if (economy >= kEconomyMin)
 81                     {
 82                         record =
 83                         {
 84                             isCompressed:   true,
 85                             data:           compr
 86                         };
 87                     }
 88                     else
 89                     {
 90                         record =
 91                         {
 92                             isCompressed:   false,
 93                             data:           data
 94                         };
 95                     }
 96                 }
 97                 else
 98                 {
 99                     record =
100                     {
101                         isCompressed:   false,
102                         data:           ""
103                     }
104                 }
105                 stor = JSON.stringify (record);
106                 this.fStore.setItem (key, stor);
107             }
108         };
109     
110         ret = declare ("TDriverCompress", [TDriver], TDriverCompress);
111     
112         return ret;
113     }
114 );
115 
116 /*
117 
118  [10]: We compress to (decompress from) a base64 string as this uses a safe character set.
119 
120  */
