488             _GetRecordSanitized: function (x)
489             {
490                 var isArray;
491                 var t;
492                 var xValue;
493                 var ret;
494 
495                 t           = typeof x;
496                 isArray     = Array.isArray (x);
497                 ret         = {};
498                 if (t === "undefined")
499                 {
500                     ret.type    = "undefined";
501                     ret.value   = "";
502                 }
503                 else if (x === null)
504                 {
505                     ret.type    = "null";
506                     ret.value   = "null";
507                 }
508                 else if (t === "object")
509                 {
510                     if (isArray)
511                     {
512                         ret.type = "array";
513                     }
514                     else
515                     {
516                         ret.type = "object";
517                     }
518                     xValue      = JSON.stringify (x);
519                     ret.value   = this._GetValueStr (xValue);
520                 }
521                 else if (t === "string")
522                 {
523                     ret.type    = "string";
524                     xValue      = this._GetValueStr (x);
525                     ret.value   = "\"" + xValue + "\"";
526                 }
527                 else if (t === "boolean")
528                 {
529                     ret.type    = "boolean";
530                     ret.value   = (x === true) ? "true" : "false"; 
531                 }
532                 else if (t === "number")
533                 {
534                     if (x == Number.POSITIVE_INFINITY)
535                     {
536                         ret.type    = "number"
537                         ret.value   = "+infinity"
538                     }
539                     else if (x == Number.NEGATIVE_INFINITY)
540                     {
541                         ret.type    = "number"
542                         ret.value   = "-infinity"
543                     }
544                     else if (isNaN (x))
545                     {
546                         ret.type    = "NaN";
547                         ret.value   = "Not a number";
548                     }
549                     else if (x % 1 === 0)
550                     {
551                         ret.type    = "integer";
552                         ret.value   = "" + x;
553                     }
554                     else if (x % 1 !== 0)
555                     {
556                         ret.type    = "float";
557                         ret.value   = "" + x;
558                     }
559                     else
560                     {
561                         ret.type    = "?number?";
562                         ret.value   = "" + x;
563                     }
564                 }
565                 else if (t === "function")
566                 {
567                     xValue      = x.toString ();
568                     ret.type    = "function";
569                     ret.value   = this._GetValueStr (xValue);
570                 }
571                 else
572                 {
573                     xValue      = JSON.stringify (x);
574                     ret.type    = "?" + t + "?";
575                     ret.value   = this._GetValueStr (xValue);
576                 }
577                 
578                 return ret;
579             },
580

583                 var ret;
584                 
585                 ret = (vStr.length <= 50) ? vStr : vStr.slice (0, 46) + "..."
586                 
587                 return ret;
588             },

633 /*
634  * [1]: We MUST define the templateString property!
635  *      Otherwise we get a mysterious error that a 'null' resource didn't load. 
636  *      During development, when we didn't specify that property, the Network 
637  *      monitor showed the following erroneous request:
638  *          Request URL: http://localhost:8383/JSDemo/null
639  *          Method: GET
640  *          Status: Request was cancelled.
641  *          
642  * [2]: Don't use XML comments in the template - the dojo parser will terminate
643  *      with an error.
644  *      
645  * [3]: Not really the place to say it here, but the more reminders the better:
646  *      On the web page including this widget, set class attribute of the body tag
647  *      to "claro": <body class="claro"> or whatever the theme demands! 
648  *      If the 'class' attribute is missing the UI elements won't be styled 
649  *      correctly, irrespective of whether the calling web page includes the 
650  *      correct stylesheets or not.
651  *      
652  * [4]: We use a closure so we don't end up with tons of variable names. UI setup is
653  *      complex, so it's good to split it up into lots of small self contained units.
654  *      
655  * [5]: A naughty hack to put a custom icon into the button. Officially, the label 
656  *      property is intended for textual labels. Thankfully the Button class accepts 
657  *      a HTML fragment as label. This saves us some css trickery with some iconClass 
658  *      and having to think about including yet another style sheet (... sigh ... dojo 
659  *      is already hard enough to learn). 
660  *      
661  * [6]: Resizeable panels don't work very well with the ace editor - We may have to make
662  *      the editor panel fixed in size and set the containing panel to overflow:auto
663  *      
664  * [7]: We have to include dojo.require, otherwise require.toUrl fails with an error
665  *      "require.toUrl: Not a function". Reason is that by the time we call this function,
666  *      dojo has 'packaged' the toUrl function inside the global require.original 
667  *      property. Requiring 'require' gives us an unpackaged require module which 
668  *      offers the toUrl function.
669  *      
670  * [10]:JSObject spec for the program used: TODO
671  * 
672  * [20]:Note that we have to turn syntax checking off - code examples
673  *      come in different languages, some of which aren't recognized by the 
674  *      editor's syntax checker (e.g. old style BASIC).
675  */
