1 /**
  2  *  @fileoverview        A dialog showing a text and a customizable set of buttons.
  3  */
  4 define 
  5 (
  6     [
  7         "dojo/_base/declare",
  8         "dijit/_WidgetBase",
  9         "dojo/Deferred",
 10         "courseware/util/validator/TValidatorJSON",
 11         "dojo/dom-construct",
 12         "dijit/Dialog",
 13         "dijit/form/Button"
 14     ],
 15     function 
 16     (
 17         declare,
 18         _WidgetBase,
 19         TDeferred,
 20         JSObjectValidator,
 21         domConstruct,
 22         TDialog,
 23         TButton
 24     )
 25     {
 26         var TButtonDialog;
 27         var ret;
 28 
 29         /**
 30          * A dialog showing a text and a set of buttons. Useful for multiple 
 31          * choice questions other than "About to reconfigure X... (OK? Cancel?)".<br/>
 32          * The dialog can be used with callbacks (one for each button's click 
 33          * event) or via the Promise API (Promise resolves with the index of the
 34          * button clicked).
 35          * <p><br/></p>
 36          * 
 37          * <b>Elements on the user interface</b>
 38          * 
 39          * <pre>
 40          * .----------------------------------------------------------------.
 41          * | Title                                                          |
 42          * |----------------------------------------------------------------|
 43          * | Question                                                       |
 44          * |                .-----------. .-----------.     .-------------. |
 45          * |                | Button[0] | | Button[1] | ... | Button[n-1] | |
 46          * |                '-----------' '-----------'     '-------------' |
 47          * '----------------------------------------------------------------'
 48          * </pre>
 49          * 
 50          * <dl>
 51          *     <dt>Title</dt>
 52          *     <dd>The dialog's title</dd>
 53          *     
 54          *     <dt>Question</dt>
 55          *     <dd>The question on which the user has to decide.</dd>
 56          *     
 57          *     <dt>Button [0] ... Button [n-1]</dt>
 58          *     <dd>The buttons representing the user's answer.</dd>
 59          * </dl>
 60          * <p><br/></p>
 61          * 
 62          * <b>Example uses</b>
 63          * @example 
 64          * // Using callbacks
 65          * 
 66          * require 
 67          * (
 68          *     [
 69          *         "courseware/gui/TButtonDialog/TButtonDialog"
 70          *     ],
 71          *     function 
 72          *     (
 73          *         TBtnDlg
 74          *     )
 75          *     {
 76          *         var dlg;
 77          *         
 78          *         dlg = new TBtnDlg
 79          *         (
 80          *             host:        window,
 81          *             buttons:
 82          *             [
 83          *                 {
 84          *                     label:       "Tell me what's 1 + 1!",
 85          *                     onClick: function ()
 86          *                     {
 87          *                         window.alert ("2");
 88          *                     }
 89          *                 },
 90          *                 {
 91          *                     label:       "I used to be decisive...",
 92          *                     onClick: function ()
 93          *                     {
 94          *                         window.alert ("... but now I'm not so sure.");
 95          *                     }
 96          *                 },
 97          *             ]
 98          *         );
 99          *         dlg.startup ();
100          *         
101          *         dlg.Show
102          *         (
103          *             "Important things box", 
104          *             "Choose today's important thing to contemplate on"
105          *         );
106          *     }
107          * );
108          * 
109          * @example
110          * // Using promise API
111          * 
112          * require 
113          * (
114          *     [
115          *         "courseware/gui/TButtonDialog/TButtonDialog"
116          *     ],
117          *     function 
118          *     (
119          *         TBtnDlg
120          *     )
121          *     {
122          *         var dlg;
123          *         
124          *         dlg = new TBtnDlg
125          *         (
126          *             host:        window,
127          *             buttons:
128          *             [
129          *                 {
130          *                     label:       "Tell me what's 1 + 1!",
131          *                     onClick: function () {}
132          *                 },
133          *                 {
134          *                     label:       "I used to be decisive...",
135          *                     onClick: function () {}
136          *                 },
137          *             ]
138          *         );
139          *         dlg.startup ();
140          * 
141          *         dlg.Show 
142          *         (
143          *             "Important things box", 
144          *             "Choose today's important thing to contemplate on"
145          *         )
146          *         .then 
147          *         {
148          *             function (decision)
149          *             {
150          *                 switch (decision)
151          *                 {
152          *                     case 0:
153          *                         window.alert ("2");
154          *                         break;
155          *                     case 1:
156          *                         window.alert ("... but now I'm not so sure.");
157          *                         break;
158          *                     default:
159          *                         window.alert ("Unknown button alert!);
160          *                 }
161          *             }
162          *         };
163          *     }
164          * );
165          * 
166          * @class       TButtonDialog
167          */
168         TButtonDialog = 
169         {
170             /**
171              * JSON schema to validate the dialog's descriptor.
172              * 
173              * @constant
174              * @type        JSON schema
175              * @private
176              */
177             kSchemaParams:
178             {
179                 "$schema":      "http://json-schema.org/draft-03/schema#",
180                 "title":        "button descriptor",
181                 "description":  "Descriptor for button row",
182                 "type":         "object",
183                 properties:
184                 {
185                     "host":
186                     {
187                         "description":      "The object hosting this button dialog",
188                         "type":             "object"
189                     },
190                     "buttons":
191                     {
192                         "description":      "Descriptors for all buttons, one button per descriptor. " +
193                                             "Buttons will be shown in the order of the descriptors.",
194                         "type":             "array",
195                         "items":
196                         {
197                             "description":      "A descriptor for a button",
198                             "type":             "object",
199                             "properties":
200                             {
201                                 "label":
202                                 {
203                                     "description":      "Text on the button, e.g. 'Cancel'",
204                                     "type":             "string"
205                                 },
206                                 "onClick":
207                                 {
208                                     "description":      "Callback (event handler) to invoke when button is clicked.",
209                                     "type":             "function"
210                                 }
211                             }
212                         }
213                     }
214                 }
215             },
216 
217             /**
218              * The descriptors for the set of buttons.
219              * 
220              * @type        JSON
221              * @private
222              */
223             fDescriptors:   null,
224             
225             /**
226              * The underlying dijit dialog.
227              * 
228              * @type        dijit/Dialog
229              * @private
230              */
231             fDialog:        null,
232             
233             /**
234              * The client using this dialog.
235              * 
236              * @type        JSObject
237              * @private
238              */
239             fHost:          null,
240             
241             /**
242              * The DOM element hosting the button set.
243              * 
244              * @type        DOMNode
245              * @private
246              */
247             fNodeButtons:   null,
248 
249             /**
250              * The DOM element hosting the question text.
251              * 
252              * @type        DOMNode
253              * @private
254              */
255             fNodeContent:   null,
256             
257             /**
258              * The promise, to use this dialog inside a <code>.then</code> construct.
259              * 
260              * @type    dojo/Deferred
261              * @private
262              */
263             fSemaphore:     null,
264         
265             /**
266              * Shows the dialog and sets title and question text. Title appears 
267              * in the dialog's title bar, question appears above the buttons.
268              * 
269              * @param       {String}    title       The dialog's title
270              * @param       {String}    question    The question for the user to decide upon.
271              * @returns     {dojo/Deferred}         A Deferred object, if the dialog is
272              *                                      to be shown using the promise API.
273              */
274             Show: function (title, question)
275             {
276                 this.fSemaphore = new TDeferred ();
277                 
278                 this.fDialog.set ("title",      title);
279                 this.fNodeContent.innerHTML = question;
280                 this.fDialog.show ();
281                 
282                 return this.fSemaphore;
283             },
284 
285             /**
286              * Dojo specific cTor.
287              * 
288              * @param {JSON}    params      The dialog's configuration. Must contain the 
289              *                              configuration for the set of buttons. Must 
290              *                              conform to {@link TButtonDialog.kSchemaParams}.
291              */
292             constructor: function (params)
293             {
294                 JSObjectValidator.AssertValid (params, this.kSchemaParams, "constructor");
295                 this.fSemaphore     = null; // of type dojo/Deferred if we want to use this dialog asynchronously.
296                 this.fDescriptors   = params.buttons;
297                 this.fHost          = params.host;
298                 this.fDialog        = null;
299                 this.fNodeContent   = null;
300                 this.fNodeButtons   = null;
301             },
302             
303             /**
304              * dTor.
305              */
306             destroy: function ()
307             {
308                 if (this.fDialog != null)
309                 {
310                     this.fDialog.destroyRecursive ();
311                 }
312             },
313 
314             /**
315              * Startup method. Sets up the dialog and the set of buttons, as 
316              * specified with the descriptor that was provided as parameter to the 
317              * constructor.
318              */
319             startup: function ()
320             {
321                 var _host = this;
322                 
323                 var i;
324                 var n;
325                 var d;
326                 var wr;
327                 var btn;
328                 
329                 /*
330                  * Create new dijit/Dialog.
331                  */
332                 this.fDialog = new TDialog
333                 (
334                     {
335                         title:      "",
336                         content:    "",
337                         style:      "width: 520px",
338                         closable:   false
339                     }
340                 );
341                 this.fDialog.startup ();
342 
343                 /*
344                  * Create DIV element to contain the question.
345                  */
346                 this.fNodeContent = domConstruct.create 
347                 (
348                     "div",
349                     {
350                         style: "width:500px"
351                     },
352                     this.fDialog.containerNode,
353                     "only"
354                 );
355             
356                 /*
357                  * Create DIV element to contain the set of buttons.
358                  */
359                 this.fNodeButtons = domConstruct.create
360                 (
361                     "div",
362                     {
363                         style: "width:500px;height:25px"
364                     },
365                     this.fDialog.containerNode,
366                     "last"
367                 );
368             
369                 /*
370                  * Create set of buttons.
371                  */
372                 n = this.fDescriptors.length;
373                 if (n >= 1)
374                 {
375                     for (i = n-1; i >= 0; i--)
376                     {
377                         d  = this.fDescriptors [i];
378                         wr = domConstruct.create
379                         (
380                             "div",
381                             {
382                                 style: "float:right"
383                             },
384                             this.fNodeButtons,
385                             "last"
386                         );
387                         btn = new TButton
388                         (
389                             {
390                                 fIndex:     i,
391                                 label:      d.label,
392                                 onClick:
393                                     function ()
394                                     {
395                                         _host.fDialog.hide ();
396                                         _host.fDescriptors[this.fIndex].onClick.call (_host.fHost); /* [10] */
397                                         _host.fSemaphore.resolve (this.fIndex);
398                                     }
399                             }
400                         );
401                         btn.startup ();
402                         domConstruct.place (btn.domNode, wr, "only");
403                     }
404                 }
405             }
406         };
407     
408         ret = declare ("TButtonDialog", [_WidgetBase], TButtonDialog);
409     
410         return ret;
411     }
412 );
413 
414 /*
415 
416 [10]: We set the context to the client who hosts this button. Therefore, a client
417       can use the 'this' reference, and be sure that 'this' always refers to that
418       client, not some other object.
419 
420  */
