116              * @return {boolean}    <code>true</code>   if the operation succeeded,
117              *                      <code>false</code>  if we weren't able to reset the program.

120             Program_DoReset: function ()
121             {
122                 var ret;                                                        /* boolean                  */
123                 
124                 ret = this.fCan_GoBackward;                                     /* [footnote] */
125                 if (this.fCan_GoBackward)
126                 {
127                     /* Get and activate scope from first position on the stack */
128                     this.fScope = this.fStack.GetFirstElement ();
129                     
130                     /* Clear the stack */
131                     this.fStack.Clear ();
132                     
133                     /* Reset state machine */
134                     this.fStateMachine.SetCurrent_State (0);
135                     
136                     /* Refresh capabilities flags */
137                     this._SetCapabilities ();
138                 }
139                 else
140                 {
141                     this._Log ("Can't reset program.");
142                 }
143 
144                 return ret;
145             },

147             /**
148              * Navigates the program forwards by one step.
149              *
150              * @return {boolean}    <code>true</code>   if the operation succeeded,
151              *                      <code>false</code>  if we weren't able to execute the next step in the program.
152              * @public
153              */
154             Program_DoStepNext: function ()
155             {
156                 var stCur;                                                      /* int                      */
157                 var sc;                                                         /* TScope                   */
158                 var trFunc;                                                     /* JS_Function              */
159                 var scrCtx;                                                     /* JS Object                */
160                 var ret;                                                        /* boolean                  */
161                 
162                 ret = this.fCan_GoForward;
163                 if (this.fCan_GoForward)
164                 {
165                     /* -------------------------------------------------------- */
166                     /* 1. Push copy of present scope onto the stack.            */
167                     /* -------------------------------------------------------- */
168                     sc = new TScope (null);
169                     sc.CopyFrom (this.fScope);
170                     this.fStack.Push (sc);
171                     
172                     /* -------------------------------------------------------- */
173                     /* 2. Run transition function within a synthetic context    */
174                     /* -------------------------------------------------------- */
175                     /* 2.1. Set up synthetic context                            */
176                     scrCtx =
177                     {
178                         globals:    {},
179                         exec:       {}
180                     };
181                     scrCtx.globals              = this.fScope.GetGlobals_AsJSObject ();
182                     scrCtx.globals._exec        = scrCtx.exec;
183                     scrCtx.globals._SetState    = function (stateNext, isTerminal) 
184                     {
185                         this._exec.state        = stateNext,
186                         this._exec.isTerminal   = isTerminal;
187                     };
188                     scrCtx.exec.state           = this.fScope.GetState          ();
189                     scrCtx.exec.isTerminal      = this.fScope.IsTerminal        ();
190                     
191                     /* 2.2. Run transition function in synthetic environment        */
192                     trFunc = this.fStateMachine.GetCurrent_TransitionFunction ();
193                     trFunc.call (scrCtx.globals);                               /* [footnote]               */
194                     
195                     /* 2.3. Copy synthetic context back into scope.             */
196                     /* 2.3.1. Reduce the globals member to contain data only.   */
197                     delete scrCtx.globals._exec;
198                     delete scrCtx.globals._SetState;
199 
200                     /* 2.3.2. Now copy what's left back into scope.             */
201                     this.fScope.SetGlobals_FromJSObject (scrCtx.globals         );
202                     this.fScope.SetState            (scrCtx.exec.state      );
203                     this.fScope.SetIsTerminal       (scrCtx.exec.isTerminal );
204                     
205                     /* -------------------------------------------------------- */
206                     /* 3. Set state in state machine                            */
207                     /* -------------------------------------------------------- */
208                     stCur = this.fScope.GetState ();
209                     this.fStateMachine.SetCurrent_State (stCur);
210                     
211                     /* -------------------------------------------------------- */
212                     /* 4. Refresh capabilities flags */
213                     /* -------------------------------------------------------- */
214                     this._SetCapabilities ();
215                 }
216                 else
217                 {
218                     this._Log ("Can't do next step.");
219                 }
220                 
221                 return ret;
222             },

463             /**
464              * Dojo specific cTor. Creates a new context, loading the given program.
465              *
466              * @param           {TDescriptor_Program}      program         The program to load.
467              */
468             constructor: function (program)
469             {
470                 var i;                                                          /* int                      */
471                 var n;                                                          /* int                      */
472                 var s;                                                          /* TDescriptor_Step         */
473                 var symList;                                                    /* TArrayList<String>       */
474                 var stpList;                                                    /* TArrayList<TDescriptor_Step> */
475                 var dict;                                                       /* TDictionary              */
476                 var iState;                                                     /* int                      */
477                 
478                 symList                 = program.GetVariables  ();
479                 dict                    = new TDictionary       ();
480                 dict.Declare (symList);
481                 
482                 this.fName             = program.GetName       ();
483                 this.fStack            = new TStack            ();
484                 this.fScope            = new TScope            (dict);
485                 this.fStateMachine     = new TStateMachine     ();
486                 
487                 stpList = program.GetSteps          ();
488                 n       = stpList.GetNumElements    ();
489                 if (n >= 1)
490                 {
491                     for (i = 0; i < n; i++)
492                     {
493                         s       = stpList.GetValue_ByIndex (i);
494                         symList = new TArrayList ();
495                         symList.AddFromArray (s.fInterestingSymbols);
496                         this.fStateMachine.AddInstruction 
497                         (
498                             s.fDemoSource,
499                             s.fComment,
500                             s.fTransitionFunction,
501                             symList
502                         );
503                     }
504                 }
505                 
506                 this._SetCapabilities ();
507                 iState =  (n >= 1) ?  0 : -1;
508                 this.fStateMachine.SetCurrent_State (iState);
509             },
510             
511             /**
512              * Logs the given message to the web browser's console.
513              *
514              * @param       {String}        msg             The message to log.
515              * @private
516              */
517             _Log: function (msg)
518             {
519                 console.log ("TContext: " + msg);
520             },

522             /**
523              * Sets the various capability flags.
524              *
525              * @private
526              */
527             _SetCapabilities: function ()
528             {
529                 var nStack;                                                     /* int                      */
530                 var sCurrent;                                                   /* int                      */
531                 var nSteps;                                                     /* int                      */
532 
533                 nStack      = this.fStack.GetNumElements ();
534                 nSteps      = this.fStateMachine.GetNumStates ();
535                 sCurrent    = this.fStateMachine.GetCurrent_State ();
536 
537                 this.fCan_GoBackward =
538                 (
539                     (nSteps   >= 1)  /* The state machine should have at least one state         */
540                  && (nStack   >= 1)  /* There needs to be at least one element on the stack      */  
541                  && (sCurrent >= 1)  /* Program pointer must be beyond the first state (# 0)     */
542                 );
543 
544                 this.fCan_GoForward = ! (this.fScope.IsTerminal ());
545             }
546         };
547     
548         ret = declare ("TContext", [], TContext);
549     
550         return ret;
551     }

554 /*
555 ---------------------------------- 
556     [fScope: null]
557         As we keep all the program's current data and state in a single container we can
558         use a stack to backtrack the demo program, thus enabling the user to 
559         navigate backwards.
560     [fStack: null]
561         We use a stack, so we can step backwards in the demo program. 
562 
563     [fCan_GoBackward: false]
564     [fCan_GoForward:  false]
565         We store capabilities as flags. These are set centrally, whenever 
566         a major event happens (e.g. 'user steps forward').
567     
568     [ret = this.fCan_...]
569         We have to store the capability flag before executing the actual step,
570         because executing the step has side effects on the capability flags.
571     
572     [trFunc.call (scrCtx.globals)]
573         We run the transition function in the context of the synthetic environment
574         we set up.
575  */
576
