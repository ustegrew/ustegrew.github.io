 10     (
 11         declare
 12     )

 14         /**
 15          * Class ID
 16          * 
 17          * @type String
 18          */
 19         var kClass = "TDriver"
 20         
 21         var TDriver;
 22         var ret;
 23 
 24         /**
 25          * <code>localStorage</code> drivers (abstract base class). Provides the 
 26          * API for the localStorage layer. Sub classes must override the 
 27          * {@link TDriver.Get} and {@link TDriver.Set} methods.
 28          * 
 29          * @class       TDriver.
 30          */
 31         TDriver = 
 32         {
 33             /**
 34              * The system's localStorage object.
 35              * 
 36              * @type window.localStorage
 37              * @private
 38              */
 39             fStore:     null,
 40             
 41             /**
 42              * Returns the value stored with the given <code>key</code>.<br/>
 43              * Please note: Concrete sub classes must override this method.
 44              * 
 45              * @param   {String}    key     The key we want to retrieve the value of.
 46              * @returns {String}    The value associated with the given <code>key</code>.
 47              */
 48             Get: function (key)
 49             {
 50                 throw kClass + "::Get (key): Must be overridden in sub class";
 51             },
 52             
 53             /**
 54              * Returns the number of data items stored in the localStore.
 55              * 
 56              * @returns {integer}       The number of stored items in the localStore.
 57              */
 58             GetSize: function ()
 59             {
 60                 var ret;
 61                 
 62                 ret = this.fStore.length;
 63                 
 64                 return ret;
 65             },
 66         
 67             /**
 68              * Returns <code>true</code> the localStorage contains a key/value 
 69              * pair with the given <code>key</code>, <code>false</code> otherwise.
 70              * 
 71              * @param   {String}    key     The key being looked up.
 72              * @returns {boolean}   <code>true</code> if a value with the given 
 73              *                      <code>key</code> exists, <code>false</code> 
 74              *                      otherwise.
 75              */
 76             HasKey: function (key)
 77             {
 78                 var ret;
 79                 
 80                 ret = this.fStore.hasOwnProperty (key);
 81                 
 82                 return ret;
 83             },
 84             
 85             /**
 86              * Removes the key/value pair with the given <code>key</code> from 
 87              * the localStorage.
 88              * 
 89              * @param   {String}    key     The key we want to remove the value of.
 90              */
 91             Remove: function (key)
 92             {
 93                 this.fStore.removeItem (key);
 94             },
 95             
 96             /**
 97              * Updates or adds the given <code>key</code>/<code>value</code> pair.<br/>
 98              * Please note: Concrete sub classes must override this method.
 99              * 
100              * @param {String}      key     The key we want to store the value as.
101              * @param {String}      value   The value to be stored.
102              */
103             Set: function (key, value)
104             {
105                 throw kClass + "::Set (key, value): Must be overridden in sub class";
106             },
107             
108             /**
109              * cTor.
110              */
111             constructor: function ()
112             {
113                 this.fStore = window.localStorage;
114             }
115         };
