1 /**
  2  *  @fileoverview        Descriptor of a jsdemo user program.. 
  3  */
  4 define 
  5 (
  6     [
  7         "dojo/_base/declare",
  8         "jsdemo/aux/storage/list/TArrayList",
  9         "jsdemo/component/program/api/TDescriptor_Step"
 10     ],
 11     function 
 12     (
 13         declare,
 14         TArrayList,
 15         TDescriptor_Step
 16     )
 17     {
 18         var TDescriptor_Program;
 19         var ret;
 20 
 21         /**
 22          *  A descriptor holding a jsdemo user program. A program is a short snippet 
 23          *  of Javascript code as an example to demonstrate some aspect of Javascript.  
 24          *  
 25          *  @class TDescriptorProgram
 26          */
 27         TDescriptor_Program = 
 28         {
 29             /**
 30              * The original record from which we create this descriptor
 31              */
 32             fOrigRecord:    null,
 33             
 34             /**
 35              * The program's name, e.g. "Example 4".
 36              * 
 37              * @type        String
 38              * @private
 39              */
 40             fName:          null,
 41             
 42             /**
 43              * The symbols for global variables used by the program, e.g. ["a", "myVar", "b"].
 44              * 
 45              * @type        TArrayList <String>  
 46              * @private
 47              */
 48             fVariables:     null,
 49             
 50             /**
 51              * The program's steps (i.e. lines of code).
 52              * 
 53              * @type        TArrayList <TDescriptor_Step>
 54              * @private
 55              */
 56             fSteps:         null,
 57             
 58             /**
 59              * Adds one step in the program.
 60              * 
 61              * @param {String}      demoSource                     The line of source code in this step.
 62              * @param {String}      comment                        An explanatory comment about this particular step.
 63              * @param {[Function]}  transitionFunction             A Javascript function we run when the user moves forward to the next step.
 64              * @param {String []}   interestingSymbols             A set of variables to show when the step has finished.
 65              * @public 
 66              */
 67             AddStep: function
 68             (
 69                 demoSource,
 70                 comment,
 71                 transitionFunction,
 72                 interestingSymbols
 73             )
 74             {
 75                 var st; 
 76                 
 77                 st = new TDescriptor_Step 
 78                 (
 79                     demoSource, 
 80                     comment, 
 81                     transitionFunction, 
 82                     interestingSymbols
 83                 );
 84                 this.fSteps.Add (st);
 85             },
 86             
 87             /**
 88              * Returns the name of this program.
 89              * 
 90              * @return {String}             The name of this program.
 91              * @public 
 92              */
 93             GetName: function () 
 94             {
 95                 return this.fName;
 96             },
 97 
 98             /**
 99              * Returns the list of global variable symbols. 
100              * 
101              * @return {TArrayList <String>}  List of global variable symbols used in this program. 
102              * @public 
103              */
104             GetVariables: function ()
105             {
106                 return this.fVariables;
107             },
108 
109             /**
110              * Returns the list of steps executed by this program.
111              * 
112              * @return TArrayList <TDescriptor_Step>      The program's steps (i.e. lines of code).
113              * @public 
114              */
115             GetSteps: function ()
116             {
117                 return this.fSteps;
118             },
119 
120             /**
121              * Dojo specific cTor.
122              * 
123              * @param   {String}    name        Name of the program, e.g. "Example 4".
124              * @param   {JSObject}  p           The program to load. 
125              * @public 
126              */
127             constructor: function (name, p)
128             {
129                 var trans;
130                 var i;
131                 var n;
132                 var t;
133                 var st;
134                 
135                 this.fOrigRecord    = p;
136                 this.fName          = name;
137                 this.fVariables     = new TArrayList ();
138                 this.fSteps         = new TArrayList ();
139                 
140                 this.fVariables.AddFromArray (p.globals.symbols);
141                 trans = p.transitionTable;
142                 n     = trans.length;
143                 if (n >= 1)
144                 {
145                     for (i = 0; i < n; i++)
146                     {
147                         t = trans [i];
148                         st = new TDescriptor_Step 
149                         (
150                             t.source, 
151                             t.comment, 
152                             t.transition, 
153                             t.symbols
154                         );
155                         this.fSteps.Add (st);
156                     }
157                 }
158             }
159         };
160     
161         ret = declare ("TDescriptor_Program", [], TDescriptor_Program);
162     
163         return ret;
164     }
165 );
