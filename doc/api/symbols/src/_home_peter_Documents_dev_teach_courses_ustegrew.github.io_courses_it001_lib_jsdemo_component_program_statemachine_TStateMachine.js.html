1 /**
  2  * @fileoverview        State machine
  3  */
  4 define 
  5 (
  6     [
  7         "dojo/_base/declare",
  8         "jsdemo/component/program/statemachine/TState",
  9         "jsdemo/aux/storage/list/TArrayList"
 10     ],
 11     function 
 12     (
 13         declare,
 14         TState,
 15         TArrayList
 16     )
 17     {
 18         var TStateMachine;
 19         var ret;
 20 
 21         /**
 22          * State machine. We use a finite state machine to model the demo
 23          * program's execution flow, and each step in the program has a 
 24          * corresponding state here.<br/>
 25          * Prior to running a program, it has to be committed to storage,
 26          * step by step. Clients do this by calling 
 27          * {@link TStateMachine.AddInstruction} for each program step. 
 28          * Upon committing a step, it will be stored as a state within 
 29          * the state machine. A state is defined as a 4-tuple with the elements:
 30          * 
 31          * <dl>
 32          *     <dt>Source</dt>
 33          *         <dd>The resp. line of Javascript code as shown in the corresponding example.</dd>
 34          *     <dt>Comment</dt>
 35          *         <dd>A comment explaining the resp. step.</dd>
 36          *     <dt>Transition function</dt>
 37          *         <dd>A Javascript function, executed in the background by the JSDemo framework. 
 38          *             The Javascript function sets the internal variables and ID for the
 39          *             next state. This provides the behaviour of the inspected program.</dd>
 40          *     <dt>Interesting symbols</dt>
 41          *         <dd>A list variables that the student might be interested to see after the transition 
 42          *             function has executed.</dd>
 43          * </dl>
 44          *
 45          * Whilst the program is running, the JSDemo framework walks the execution path as determined by 
 46          * the stored transition functions. Each transition function is responsible to set the ID of the 
 47          * next state to be visited. We just use ordinary Javascript functions; it is overkill to design 
 48          * a separate execution framework. Instead we provide a minimal enviroment for the functions, with
 49          * simple access to the program's global variables via <i>this</i> reference 
 50          * (<code>this.<variable></code>) and one function to set the next state 
 51          * (<code>this._SetState (id, isterminal)</code>). This small environment will be set up dynamically
 52          * whilst the program is running. All this is happening in the {@link TContext} class.
 53          *
 54          * <br/>TODO: Code to generate setup of enviroment for javascript transition function.
 55          *  
 56          * @class               TStateMachine
 57          * @see                 TState
 58          * @see                 TContext
 59          */
 60         TStateMachine = 
 61         {
 62             /**
 63              * List of source code lines of the program.
 64              * 
 65              * @type        TArrayList <{@link String}>
 66              * @private
 67              */
 68             fSourceCodeAll:     null,                                           /* private TArrayList<String>       */
 69             
 70             /**
 71              * Index to the state we currently look at. 
 72              * 
 73              * @type        int
 74              * @private
 75              */
 76             fStateCurrentID:      null,                                         /* private int                      */
 77 
 78             /**
 79              * The current state object (4-tuple)
 80              * 
 81              * @type        TState
 82              * @private
 83              */
 84             fStateCurrentTuple:    null,
 85 
 86             /**
 87              * List of all states (program steps).
 88              * 
 89              * @type        TArrayList <{@link TState}>
 90              * @private
 91              */
 92             fStates:            null,                                           /* private TArrayList <TState>      */
 93 
 94             /**
 95              * Commits a state (step) to the storage. 
 96              * 
 97              * @param       {String}                        source              The line of source code as shown in the example.
 98              * @param       {String}                        comment             An explanatory comment.
 99              * @param       {[Function]}                    transFunc           A Javascript function, executed behind the scenes.
100              * @param       {TArrayList <String>}     interestingSymbols  Names of interesting variables (symbols) at this current state.
101              * @public
102              */
103             AddInstruction: function
104             (
105                 source,
106                 comment,
107                 transFunc,
108                 interestingSymbols
109             )
110             {
111                 var st;                                                         /* TState                           */
112                 
113                 st = new TState (source, comment, transFunc, interestingSymbols);
114                 this.fStates.Add (st);
115                 this.fSourceCodeAll.Add (source);
116             },
117            
118             /**
119              * Returns all source code lines of the demo program as an indexed list.
120              * 
121              * @return      {TArrayList <String>}                         All source code lines of the demo program.
122              * @public
123              */
124             GetAll_SourceLines: function ()
125             {
126                 var ret;                                                        /* TArrayList<String>               */
127                 
128                 ret = new TArrayList ();
129                 ret.AddFromArrayList (this.fSourceCodeAll);
130                 
131                 return ret;
132             },
133            
134             /**
135              * Returns the explanatory comment for the current state (step).
136              * 
137              * @return      {String}                                            Explanatory comment for the current state.
138              * @public
139              */
140             GetCurrent_Comment: function ()
141             {
142                 var s;                                                          /* TState                           */
143                 var ret;                                                        /* String                           */
144                
145                 s   = this._GetCurrentState ();
146                 ret = s.GetComment ();
147                
148                 return ret;
149             },
150            
151             /**
152              * Returns the names of interesting variables (symbols) at the current state.
153              * 
154              * @return  {TArrayList <String>}                             The names of interesting variables (symbols) at this current state.
155              * @public
156              */
157             GetCurrent_InterestingSymbols: function ()
158             {
159                 var s;                                                          /* TState                           */
160                 var ret;                                                        /* TArrayList<String>               */
161                 
162                 s   = this._GetCurrentState ();
163                 ret = s.GetInterestingSymbols ();
164                 
165                 return ret;
166             },
167             
168             /**
169              * Returns the source code for the current state.
170              *
171              * @return  {String}                                                Source code for the current state.
172              * @public
173              */
174             GetCurrent_Source: function ()
175             {
176                 var s;                                                          /* TState                           */
177                 var ret;                                                        /* String                           */
178          
179                 s   = this._GetCurrentState ();
180                 ret = s.GetSourceCode ();
181                 
182                 return ret;
183             },
184             
185             /**
186              * Returns the ID of the current state. This maps directly
187              * to the line number of the modelled program.
188              * 
189              * @return      {int}                                               The ID of the current state.
190              * @public
191              */
192             GetCurrent_State: function ()
193             {
194                 return this.fStateCurrentID;
195             },
196             
197             /**
198              * Returns the current state's transition function.
199              * 
200              * @return      {Function}                                          The current state's transition function
201              * @public
202              */
203             GetCurrent_TransitionFunction: function ()
204             {
205                 var s;                                                          /* TState                           */
206                 var ret;                                                        /* [Function]                       */
207                 
208                 s   = this._GetCurrentState ();
209                 ret = s.GetTransitionFunction ();
210                 
211                 return ret;
212             },
213            
214             /**
215              * Returns the number of stored states.
216              * 
217              * @return      {int}                                               The number of stored states.
218              * @public
219              */
220             GetNumStates: function ()
221             {
222                 var ret;                                                        /* int                              */
223                 
224                 ret = this.fStates.GetNumElements ();
225                 
226                 return ret;
227             },
228             
229             /**
230              * Sets the next state to be visited. The given ID corresponds to the resp. line of source code
231              * with 0 (zero) being the first line, 1 (one) the second, 2 the third, ...
232              *
233              * @param       {int}           i           The ID of the state to be set.
234              * @throws      {OutOfBoundsException}      If given index is out of bounds.
235              * @public
236              */
237             SetCurrent_State: function (i)
238             {
239                 this.fStateCurrentID       = i;
240                 this.fStateCurrentTuple    = this.fStates.GetValue_ByIndex (i);
241             },
242            
243             /**
244              * Returns the current state 4-tuple.
245              * 
246              * @return                                  The current state object.
247              * @throws      {NullPointerException}      If no current state has been set yet.
248              * @private
249              */
250             _GetCurrentState: function ()
251             {
252                 var ret;                                                        /* TState                           */
253                 
254                 ret = this.fStateCurrentTuple;
255                 if (ret == null)
256                 {
257                     throw "Current state hasn't been set yet. Did you add any program instructions (AddInstruction (...))?";
258                 }
259                 
260                 return ret;
261             },
262  
263             /**
264              * Dojo specific cTor.
265              */
266             constructor: function ()
267             {
268  
269                 this.fStates            = new TArrayList ();
270                 this.fStateCurrentID    = 0;
271                 this.fStateCurrentTuple = null;
272                 this.fSourceCodeAll     = new TArrayList ();
273             }
274         };
275     
276         ret = declare ("TStateMachine", [], TStateMachine);
277     
278         return ret;
279     }
280 );
