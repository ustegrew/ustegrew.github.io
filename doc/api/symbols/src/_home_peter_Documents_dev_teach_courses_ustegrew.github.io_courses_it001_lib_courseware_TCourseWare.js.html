397             /**
398              * Validates the given course descriptor. If the descriptor fails validation
399              * we will throw an exception with a detailed error message.
400              * 
401              * @param       {JSON string}   descriptor      The descriptor to be validated.
402              * @throws      {string}        An exception (message containing details) if the given descriptor wasn't valid. 
403              */
404             _AssertDescriptorValid: function (descriptor)
405             {
406                 var i;
407                 var n;
408                 var d;
409                 var d;
410                 var dS;
411                 var result;
412                 var followupSchema;
413                 
414                 jsonValidator.AssertValid (descriptor, this.kSchemaDescriptorGlobal, "TCourseWare::LoadCourse(kSchemaDescriptorGlobal)");
415                 n      = descriptor.length;
416                 if (n >= 1)
417                 {
418                     for (i = 0; i < n; i++)
419                     {
420                         d  = descriptor [i];
421                         c  = d.category;
422                         if (c === "folder")
423                         {
424                             followupSchema = this.kSchemaDescriptorNodeFolder;
425                         }
426                         else if (c === "lesson")
427                         {
428                             followupSchema = this.kSchemaDescriptorNodeLesson;
429                         }
430                         else if (c === "external")
431                         {
432                             followupSchema = this.kSchemaDescriptorNodeExtRes;
433                         }
434                         else
435                         {
436                             dS = JSON.stringify (d, null, 4);
437                             throw "TCourseWare::_AssertDescriptorValid: Unknown node type '" + c + "' in node:\n" + dS;
438                         }
439                         jsonValidator.AssertValid (d, followupSchema, "TCourseWare::LoadCourse(followupSchema)");
440                     }
441                 }
442             },

444             /**
445              * An event handler executor. Executes one of the TCourseWare module wide 
446              * events (defined in the descriptor which was loaded during execution of
447              * the {@link #LoadCourse} method. 
448              * 
449              * Executes the given function in context 'window'. The provided 
450              * function name is assumed to be a string containing a dot separated 
451              * path:
452              * 
453              *     <code>courseName dot handlerName</code>
454              *     
455              * (e.g. 'it001.onLoadCourse'). This maps to an object hierarchy within the 
456              * global <code>window</code>: <code>window.courseName.handlerName</code>
457              * which means that this object hierarchy has to be set up before any of the
458              * event handlers is called.
459              * 
460              * @param   {string}    fn      The handler to be called.
461              */
462             _EventExec: function (fn)
463             {
464                 var path;
465                 var i;
466                 var n;
467                 var o;
468                 var p;
469 
470                 
471                 path = fn.split (".");                                          /* [11] */
472                 n    = path.length;
473                 o    = window;
474                 if (n >= 1)
475                 {
476                     for (i = 0; i < n; i++)
477                     {
478                         p = path [i];
479                         o = o [p];
480                     }
481                 }
482                 else
483                 {
484                     o = null;
485                 }
486                 
487                 if (o !== null)
488                 {
489                     if (typeof o === "function")
490                     {
491                         o.call (window);
492                     }
493                 }
494             },

496             /**
497              * Worker method to load a course into the environment. Clients must 
498              * provide a JSON record describing the course.
499              * 
500              * @param       {type} descriptor   The JSON record describing the course.
501              *                                  Must comply with the JSON schema 
502              *                                  defined in {@link #kSchemaDescriptorGlobal}.
503              */
504             _LoadCourse: function (descriptor)
505             {
506                 var _this = this;
507                 
508                 var urls;
509                 var i;
510                 var n;
511                 var url;
512                 var lnk;
513                 var headNode;
514                 
515                 /* 1. Validate descriptor */
516                 this._AssertDescriptorValid (descriptor);
517 
518                 /* 2. Retrieve event handler names for the functions covering events: 
519                  *     'course loaded', 
520                  *     'article loaded',
521                  *     'article unloaded'.                                      [30]
522                  */
523                 this.fHandlerOnLoadCourse       = descriptor.scripts.onLoadCourse;
524                 this.fHandlerOnLoadArticle      = descriptor.scripts.onLoadArticle;
525                 this.fHandlerOnUnloadArticle    = descriptor.scripts.onUnloadArticle;
526                 
527                 /* 3. Load custom style sheets and scripts */
528                 /* 3.1. Load custom stylesheets into the page header.                [20] */
529                 headNode    = query ("head");
530                 urls        = descriptor.stylesheets;
531                 n           = urls.length;
532                 if (n >= 1)
533                 {
534                     for (i = 0; i < n; i++)
535                     {
536                         url = urls [i];
537                         lnk = '<link rel="stylesheet" type="text/css" href="' + url + '">';
538                         headNode.append (lnk);
539                     }
540                 }
541 
542                 /* 3.2. Load custom scripts and execute onLoadCourse event handler
543                  *      after all scripts have loaded
544                  */
545                 urls = descriptor.scripts.resources;
546                 n    = urls.length;
547                 if (n >= 1)
548                 {
549                     require 
550                     (
551                         urls,
552                         function ()
553                         {
554                             _this.fGUI.LoadCourse (descriptor);
555                             _this._EventExec (_this.fHandlerOnLoadCourse);
556                             _this.fGUI.LoadInitialArticle ();
557                         }
558                     );
559                 }
560             },
561             
562         };

565     
566         return ret;
567     }
