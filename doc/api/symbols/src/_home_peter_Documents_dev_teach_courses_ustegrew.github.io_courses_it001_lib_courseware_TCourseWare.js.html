497             /**
498              * Validates the given course descriptor. If the descriptor fails validation
499              * we will throw an exception with a detailed error message.
500              * 
501              * @param       {JSON string}   descriptor      The descriptor to be validated.
502              * @throws      {string}        An exception (message containing details) if the given descriptor wasn't valid. 
503              */
504             _AssertDescriptorValid: function (descriptor)
505             {
506                 var i;
507                 var n;
508                 var d;
509                 var d;
510                 var dS;
511                 var result;
512                 var followupSchema;
513                 
514                 jsonValidator.AssertValid (descriptor, this.kSchemaDescriptorGlobal, "TCourseWare::LoadCourse(kSchemaDescriptorGlobal)");
515                 n      = descriptor.length;
516                 if (n >= 1)
517                 {
518                     for (i = 0; i < n; i++)
519                     {
520                         d  = descriptor [i];
521                         c  = d.category;
522                         if (c === "folder")
523                         {
524                             followupSchema = this.kSchemaDescriptorNodeFolder;
525                         }
526                         else if (c === "lesson")
527                         {
528                             followupSchema = this.kSchemaDescriptorNodeLesson;
529                         }
530                         else if (c === "external")
531                         {
532                             followupSchema = this.kSchemaDescriptorNodeExtRes;
533                         }
534                         else
535                         {
536                             dS = JSON.stringify (d, null, 4);
537                             throw "TCourseWare::_AssertDescriptorValid: Unknown node type '" + c + "' in node:\n" + dS;
538                         }
539                         jsonValidator.AssertValid (d, followupSchema, "TCourseWare::LoadCourse(followupSchema)");
540                     }
541                 }
542             },

544             /**
545              * An event handler executor. Executes one of the TCourseWare module wide 
546              * events (defined in the descriptor which was loaded during execution of
547              * the {@link #LoadCourse} method. 
548              * 
549              * Executes the given function in context 'window'. The provided 
550              * function name is assumed to be a string containing a dot separated 
551              * path:
552              * 
553              *     <code>courseName dot handlerName</code>
554              *     
555              * (e.g. 'it001.onLoadCourse'). This maps to an object hierarchy within the 
556              * global <code>window</code>: <code>window.courseName.handlerName</code>
557              * which means that this object hierarchy has to be set up before any of the
558              * event handlers is called.
559              * 
560              * @param   {string}    fn      The handler to be called.
561              */
562             _EventExec: function (fn)
563             {
564                 var path;
565                 var i;
566                 var n;
567                 var o;
568                 var p;
569 
570                 
571                 path = fn.split (".");                                          /* [11] */
572                 n    = path.length;
573                 o    = window;
574                 if (n >= 1)
575                 {
576                     for (i = 0; i < n; i++)
577                     {
578                         p = path [i];
579                         o = o [p];
580                     }
581                 }
582                 else
583                 {
584                     o = null;
585                 }
586                 
587                 if (o !== null)
588                 {
589                     if (typeof o === "function")
590                     {
591                         o.call (window);
592                     }
593                 }
594             },

596             /**
597              * Worker method to load a course into the environment. Clients must 
598              * provide a JSON record describing the course.
599              * 
600              * @param       {type} descriptor   The JSON record describing the course.
601              *                                  Must comply with the JSON schema 
602              *                                  defined in {@link #kSchemaDescriptorGlobal}.
603              */
604             _LoadCourse: function (descriptor)
605             {
606                 var _this = this;
607                 
608                 var urls;
609                 var i;
610                 var n;
611                 var url;
612                 var lnk;
613                 var headNode;
614                 
615                 /* 1. Validate descriptor */
616                 this._AssertDescriptorValid (descriptor);
617 
618                 /* 2. Retrieve event handler names for the functions covering events: 
619                  *     'course loaded', 
620                  *     'article loaded',
621                  *     'article unloaded'.                                      [30]
622                  */
623                 this.fHandlerOnLoadCourse       = descriptor.scripts.onLoadCourse;
624                 this.fHandlerOnLoadArticle      = descriptor.scripts.onLoadArticle;
625                 this.fHandlerOnUnloadArticle    = descriptor.scripts.onUnloadArticle;
626                 
627                 /* 3. Load custom style sheets and scripts */
628                 /* 3.1. Load custom stylesheets into the page header.                [20] */
629                 headNode    = query ("head");
630                 urls        = descriptor.stylesheets;
631                 n           = urls.length;
632                 if (n >= 1)
633                 {
634                     for (i = 0; i < n; i++)
635                     {
636                         url = urls [i];
637                         lnk = '<link rel="stylesheet" type="text/css" href="' + url + '">';
638                         headNode.append (lnk);
639                     }
640                 }
641 
642                 /* 3.2. Load custom scripts and execute onLoadCourse event handler
643                  *      after all scripts have loaded
644                  */
645                 urls = descriptor.scripts.resources;
646                 n    = urls.length;
647                 if (n >= 1)
648                 {
649                     require 
650                     (
651                         urls,
652                         function ()
653                         {
654                             _this.fGUI.LoadCourse (descriptor);
655                             _this._EventExec (_this.fHandlerOnLoadCourse);
656                             _this.fGUI.LoadInitialArticle ();
657                         }
658                     );
659                 }
660             },
661             
662         };
663     
664         ret = declare ("TCourseWare", [_WidgetBase], TCourseWare);
665     
666         return ret;
667     }
