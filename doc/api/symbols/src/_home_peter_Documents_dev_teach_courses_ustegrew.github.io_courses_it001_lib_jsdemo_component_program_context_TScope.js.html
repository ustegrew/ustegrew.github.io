1 /**
  2  * @fileoverview        Data storage and state tracking.
  3  */
  4 define 
  5 (
  6     [
  7         "dojo/_base/declare",
  8         "jsdemo/aux/storage/map/TDictionary"
  9     ],
 10     function 
 11     (
 12         declare,
 13         TDictionary
 14     )
 15     {
 16         var TScope;
 17         var ret;
 18 
 19         /**
 20          * The set of all entities that are visible within one particular step in the demo program. Contains 
 21          * the global variables and state ID of the program at that point in time. Since we can store scope
 22          * objects we can keep track of the program's behaviour over time. This is particularly important 
 23          * for allowing the user to step backwards through the program.
 24          *
 25          * @class   TScope
 26          */
 27         TScope = 
 28         {
 29             /**
 30              * The global variables.
 31              * 
 32              * @type        TDictionary
 33              * @private
 34              */
 35             fGlobals:               null,
 36 
 37             /**
 38              * A flag to denote whether the current step is a terminal step.
 39              * 
 40              * @type        boolean
 41              * @private
 42              */
 43             fIsTerminal:            null,
 44 
 45             /**
 46              * The ID of the current state.
 47              * 
 48              * @type        int
 49              * @private
 50              */
 51             fState:                 null,
 52 
 53             /**
 54              * Copies data from the given scope into this one. Any previous data stored in this 
 55              * scope object will be lost.
 56              *
 57              * @param       {TScope}    other       The scope from which we copy.
 58              * @public
 59              */
 60             CopyFrom: function (other)
 61             {
 62                 this.fGlobals = new TDictionary ();
 63                 this.fGlobals.CopyFrom (other.fGlobals);
 64                 this.fState = other.fState;
 65                 this.fIsTerminal = other.fIsTerminal;
 66             },
 67             
 68             /**
 69              * Returns the globals as JS Object.
 70              * 
 71              * @returns {JSObject}      The globals as JS Object.
 72              */
 73             GetGlobals_AsJSObject : function ()
 74             {
 75                 var ret;
 76                 
 77                 ret = this.fGlobals.GetAsJSObject ();
 78                 
 79                 return ret;
 80             },
 81             
 82             /**
 83              * Returns the name of the global variable #<code>i</code>.
 84              * 
 85              * @param       {int}       i               The index to the variable we request the name of. The first variable has index 0 (zero).
 86              * @return      {String}                    The name of the requested variable.
 87              * @throws      {OutOfBoundsException}      If given index is out of bounds.
 88              * @public  
 89              */
 90             GetGlobals_Key: function (i)
 91             {
 92                 var ret;                                                        /* String */
 93 
 94                 ret = this.fGlobals.GetKey (i);
 95 
 96                 return ret;
 97             },
 98 
 99             /**
100              * Returns the number of global variables.
101              * 
102              * @return      {int}               The number of global variables.
103              * @public  
104              */
105             GetGlobals_NumEntries: function ()
106             {
107                 var ret;                                                        /* int */
108 
109                 ret = this.fGlobals.GetNumEntries ();
110 
111                 return ret;
112             },
113 
114             /**
115              * Returns the value of the global variable #<code>i</code>. 
116              * 
117              * @param       {int}       i               The index to the variable we request the value of. The first variable has index 0 (zero).
118              * @return      {Any}                       The value of the requested variable.
119              * @throws      {OutOfBoundsException}      If given index is out of bounds.
120              * @public  
121              */
122             GetGlobals_Value_ByIndex: function (i)
123             {
124                 var ret;                                                        /* Any */
125 
126                 ret = this.fGlobals.GetValue_ByIndex (i);
127 
128                 return ret;
129             },
130 
131             /**
132              * Returns the value of the global variable <code>[key]</code>. 
133              *  
134              * @param       {int}       name        The name of the variable we request the value of.
135              * @return      {String}                The value of the requested variable.
136              * @throws      {NoSuchKeyException}    If given key does not exist.
137              * @public  
138              */
139             GetGlobals_Value_ByKey: function (name)
140             {
141                 var ret;                                                        /* Any */
142 
143                 ret = this.fGlobals.GetValue_ByKey (name);
144 
145                 return ret;
146             },
147 
148             /**
149              * Returns a copy of a subset of all global variables, as an indexed list (index is zero based).
150              * 
151              * @param       {TArrayList <String>}     keys    The names of the variables we are interested in.
152              * @return      {String}                                The value of the requested variable.
153              * @throws      {NoSuchKeyException}                    If one or more of the given keys does not exist.
154              * @public  
155              */
156             GetGlobals_Values: function (keys)
157             {
158                 var i;                                                          /* int                      */
159                 var n;                                                          /* int                      */
160                 var k;                                                          /* String                   */
161                 var x;                                                          /* Any                      */
162                 var ret;                                                        /* TArrayList<Any>          */
163 
164                 ret = new TArrayList ();
165                 n   = keys.GetNumElements ();
166                 if (n >= 1)
167                 {
168                     for (i = 0; i < n; i++)
169                     {
170                         k = keys.get (i);
171                         x = this.fGlobals.GetValue_ByKey (k);
172                         ret.add (x);
173                     }
174                 }
175 
176                 return ret;
177             },
178 
179             /**
180              * Returns the ID of the state associated with this scope. 
181              * 
182              * @return          {int}       The current step ID.
183              * @public
184              */            
185             GetState: function ()
186             {
187                 return this.fState;
188             },
189 
190             /**
191              * Returns <code>true</code> if the associated state is a final state,
192              * <code>false</code> otherwise.
193              * 
194              * @return          {boolean}       <code>true</code> if the associated state is final,
195              *                                  <code>false</code> otherwise.
196              * @public
197              */            
198             IsTerminal: function ()
199             {
200                 return this.fIsTerminal;
201             },
202             
203             /**
204              * Sets global variables from the given JS Object.
205              * 
206              * @param   {JSObject} record       The record from which to set the globals.
207              * @public
208              */
209             SetGlobals_FromJSObject: function (record)
210             {
211                 this.fGlobals.SetFromJSObject (record);
212             },
213 
214             /**
215              * Sets the value of the global variable #<code>i</code> to the given value (<code>value</code>).
216              * 
217              * @param       {int}       i               The index to the variable we set. The first variable has index 0 (zero).
218              * @param       {Any}       value           Value we set the variable to.
219              * @throws      {OutOfBoundsException}      If given index is out of bounds.
220              * @public
221              */            
222             SetGlobals_Value_ByIndex: function (i, value)
223             {
224                 this.fGlobals.SetValue (i, value);
225             },
226 
227             /**
228              * Sets the value of the variable with the given name (<code>k</code>) to the given value (<code>value</code>).
229              * 
230              * @param       {String}    k           Name of variable we set.
231              * @param       {Any}       value       Value we set the variable to.
232              * @throws      {NoSuchKeyException}    If given key does not exist.
233              * @public
234              */            
235             SetGlobals_Value_ByKey: function (k, value)
236             {
237                 this.fGlobals.SetValue (key, value);
238             },
239 
240             /**
241              * (Re)sets the <i>terminal</i> flag of the associated state.
242              * 
243              * @param   {boolean}   isTerminal  <code>true</code>  if the state is marked terminal,
244              *                                  <code>false</code> otherwise.
245              * @public
246              */            
247             SetIsTerminal: function (isTerminal)
248             {
249                 this.fIsTerminal = isTerminal;
250             },
251 
252             /**
253              * Sets the ID and Re)sets the <i>terminal</i> flag of the associated state. 
254              * In effect, setting the ID sets the number of the associated line in the source code.
255              * 
256              * @param   {int}       st          The state's ID.
257              * @param   {boolean}   isTerminal  <code>true</code>  if the state is marked terminal,
258              *                                  <code>false</code> otherwise.
259              * @public
260              */            
261             SetState: function (st, isTerminal)
262             {
263                 this.fState         = st;
264                 this.fIsTerminal    = isTerminal;
265             },
266 
267             /**
268              * Dojo specific cTor. 
269              *
270              * @param   {TDictionary}   globals     The demo program's global variables.
271              */
272             constructor: function (globals)
273             {
274                 this.fGlobals       = globals;
275                 this.fState         = 0;
276                 this.fIsTerminal    = false;
277             }
278         };
279     
280         ret = declare ("TScope", [], TScope);
281     
282         return ret;
283     }
284 );
