438             _AssertKnownCategory: function (c)
439             {
440                 var kCats = ["folder", "lesson", "external"];
441                 
442                 var i;
443                 
444                 i = kCats.indexOf (c);
445                 if (i <= -0.5)                                                  /* [10] */
446                 {
447                     throw "TCourseWareGUI::_AssertKnownCategory: Unknown category '" + 
448                           c + 
449                           "' in descriptor: " +
450                           JSON.stringify (descriptor, null, 4);
451                 }

454             _GetDescriptor: function (query)
455             {
456                 var queryStr;
457                 var repoStr;
458                 var items;
459                 var isArray;
460                 var n;
461                 var x;
462                 var ret;
463                 
464                 queryStr   = JSON.stringify (query, null, 4);
465                 repoStr    = JSON.stringify (this.fTree.model.store, null, 4);
466                 items      = this.fTree.model.store.query (query);
467                 isArray    = Array.isArray (items);
468                 if (! isArray)
469                 {
470                     throw "TCourseWareGUI::_LoadArticle: Can't find article for query '" + 
471                           queryStr + 
472                           "' in repository:\n" +
473                           repoStr;
474                 }
475                 
476                 n = items.length;
477                 if (n <= 0)
478                 {
479                     throw "TCourseWareGUI::_LoadArticle: Can't find article for query '" + 
480                           queryStr + 
481                           "' in repository:\n" +
482                           repoStr;
483                 }
484                 
485                 ret = items[0];
486 
487                 return ret;
488             },

497             {

503                 this._AssertKnownCategory (cat);

508                     this.fWorksheet.RequestTerminate().then
509                     (
510                         function ()

515                         }
516                     );
517                 }
518                 else
519                 {
520                     d.resolve ();
521                 }
522                 
523                 d.then
524                 (
525                     function ()
526                     {
527                         if (cat === "folder")                                           /* [11] */
528                         {
529                             _this._LoadFolder (descriptor);
530                         }
531                         else if (cat === "lesson")
532                         {
533                             _this._LoadLesson (descriptor);
534                         }
535                         else if (cat === "external")
536                         {
537                             _this._LoadExternal (descriptor);
538                         }
539                     }

562                 this.fPnlArticle.set ("title", item.heading);

564                 (
565                     function ()
566                     {
567                         return _this.fPnlExercises.set  ("href",  item.urlExercises);
568                     }
569                 ).then
570                 (
571                     function () 
572                     {
573                         return _this._Worksheet_Initialize ();
574                     }
575                 ).then
576                 (
577                     function () 
578                     {
579                         return _this.fPnlExercises.set ("open", false);
580                     }
581                 ).then
582                 (
583                     function () 
584                     {
585                         _this._NotifyFinishedLoadArticle ();

685     [11]:   I could equally well use a switch statement - however, using strings in
