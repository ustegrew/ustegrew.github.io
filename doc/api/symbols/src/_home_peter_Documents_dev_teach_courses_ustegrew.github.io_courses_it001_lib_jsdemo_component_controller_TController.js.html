1 /**
  2  * @fileoverview        Controller. Connects UI, background logic and data model. 
  3  */
  4 define 
  5 (
  6     [
  7         "dojo/_base/declare"
  8     ],
  9     function 
 10     (
 11         declare
 12     )
 13     {
 14         var TController;
 15         var ret;
 16             
 17         /**
 18          * The controller. Connects UI, background logic and data model. 
 19          * 
 20          * @class       TController
 21          */
 22         TController =
 23         {
 24             /**
 25              * The hosting JSDemo component.
 26              * 
 27              * @type    TJSDemo
 28              * @private
 29              */
 30             fHost: null,
 31             
 32             /**
 33              * The demo program executor (background logic).
 34              * 
 35              * @type    TProgram
 36              * @private
 37              */
 38             fProgram:           null,
 39             
 40             /**
 41              * The user interface.
 42              * 
 43              * @type    TGUI
 44              * @private
 45              */
 46             fUI:                null,
 47             
 48             NotifyInitFinished: function ()
 49             {
 50                 this.fHost.NotifyInitFinished ();
 51             },
 52             
 53             /**
 54              * Notification: Load a demo program.
 55              * 
 56              * @param   {TDescriptor_Program}   pd      The program to load. <div style="color:red;"><b>TODO</b>: Write docu about program format.</div>
 57              * @public
 58              */
 59             Notify_LoadProgram: function (pd)
 60             {
 61                 this.fProgram.Load (pd);
 62                 this._UpdateUI ();
 63             },
 64             
 65             /**
 66              * Notification: Navigate one step backwards in the demo program.
 67              * 
 68              * @public
 69              */
 70             Notify_Navigate_Backwards: function ()
 71             {
 72                 this.fProgram.DoNavBackward ();
 73                 this._UpdateUI ();
 74             },
 75             
 76             /**
 77              * Notification: Navigate one step forwards in the demo program.
 78              * 
 79              * @public
 80              */
 81             Notify_Navigate_Forwards: function ()
 82             {
 83                 this.fProgram.DoNavForward ();
 84                 this._UpdateUI ();
 85             },
 86             
 87             /**
 88              * Notification: Reset the demo program, i.e. go back to the start state.
 89              * 
 90              * @public
 91              */
 92             Notify_Navigate_Reset: function ()
 93             {
 94                 this.fProgram.DoNavReset ();
 95                 this._UpdateUI ();
 96             },
 97             
 98             /**
 99              * Sets the hosting controller. Has to be done in a separate 
100              * setter method, rather than the constructor (otherwise the
101              * Dojo framework will flag up 'undefined' reference errors.
102              * 
103              * @param   {TJSDemo}   host    The hosting JSDemo component.
104              * @public
105              */
106             SetHost: function (host)
107             {
108                 this.fHost = host;
109             },
110             
111             /**
112              * Sets UI and background logic. JS demo programs can then be loaded via {@link TController.LoadProgram}.
113              * 
114              * @param {TProgram}        prog    Background logic, executing the demo program.
115              * @param {TGUI}            ui      User interface (Panel).
116              * @public
117              */
118             SetSubjects: function (prog, ui)
119             {
120                 this.fProgram   = prog;
121                 this.fUI        = ui;
122             },
123             
124             /**
125              * Dojo specific cTor.
126              */
127             constructor: function ()
128             {
129             },
130             
131             /**
132              * Updates the user interface, e.g. when a demo program step has been computed.
133              * 
134              * @private
135              */
136             _UpdateUI: function ()
137             {
138                 var snapshot;
139                 
140                 snapshot = this.fProgram.GetSnapshot ();
141                 this.fUI.Notify_SetStatus (snapshot);
142             }
143         };
144 
145         ret = declare ("TController", [], TController);
146 
147         return ret;
148     }
149 );
