1 /**
  2  *  @fileoverview        A text box
  3  */
  4 define 
  5 (
  6     [
  7         "dojo/_base/declare",
  8         "dijit/_WidgetBase",
  9         "dojo/Deferred",
 10         "dojo/dom-construct",
 11         "dijit/Dialog"
 12     ],
 13     function 
 14     (
 15         declare,
 16         _WidgetBase,
 17         TDeferred,
 18         domConstruct,
 19         TDialog
 20     )
 21     {
 22         var TTextWindow;
 23         var ret;
 24 
 25         /**
 26          * A text box with heading and copyable text area below. Used for the 
 27          * "Export to clipboard" feature.
 28          * 
 29          * @class       TTextWindow
 30          */
 31         TTextWindow = 
 32         {
 33             /**
 34              * 
 35              * 
 36              * @type
 37              * @private
 38              */
 39             fDialog: null,
 40             
 41             /**
 42              * 
 43              * 
 44              * @type
 45              * @private
 46              */
 47             fNodeHeading: null,
 48             
 49             /**
 50              * 
 51              * 
 52              * @type
 53              * @private
 54              */
 55             fNodeText: null,
 56             
 57             /**
 58              * 
 59              * 
 60              * @type
 61              * @private
 62              */
 63             fSemaphore: null,
 64             
 65             /**
 66              * 
 67              * 
 68              * @type
 69              * @private
 70              */
 71             Show: function (title, heading, text)
 72             {
 73                 var _this = this;
 74                 
 75                 this.fSemaphore = new TDeferred ();
 76                 
 77                 this.fDialog.set ("title",  title);
 78                 this.fNodeHeading.innerHTML = heading;
 79                 this.fNodeText.value        = text;
 80                 
 81                 this.fDialog.show().then
 82                 (
 83                     function ()
 84                     {
 85                         _this.fNodeText.scrollTop = 0;
 86                     }
 87                 );
 88             
 89                 return this.fSemaphore;
 90             },
 91             
 92             /**
 93              * Dojo specific cTor.
 94              */
 95             constructor: function ()
 96             {
 97                 this.fDialog        = null;
 98                 this.fNodeHeading   = null;
 99                 this.fNodeText      = null;
100             },
101             
102             destroy: function ()
103             {
104                 if (this.fDialog != null)
105                 {
106                     this.fDialog.destroyRecursive ();
107                 }
108             },
109 
110             /* -------------------------------------------------------------
111              * Dijit overrides 
112              * ------------------------------------------------------------- */
113         
114             /**
115              * Startup method (for widgets). This overrides the _WidgetBase::startup ().
116              * 
117              * Excerpt, Dojo documentation:
118              *     + postCreate
119              *          This is typically the workhorse of a custom widget. The 
120              *          widget has been rendered (but note that child widgets in 
121              *          the containerNode have not!). The widget though may not 
122              *          be attached to the DOM yet so you shouldnâ€™t do any sizing 
123              *          calculations in this method.
124              *     
125              *     + startup
126              *          If you need to be sure parsing and creation of any child 
127              *          widgets has completed, use startup. This is often used 
128              *          for layout widgets like BorderContainer. If the widget 
129              *          does JS sizing, then startup() should call resize(), 
130              *          which does the sizing.
131              */
132             startup: function ()
133             {
134                 var _this = this;
135                 
136                 this.fDialog = new TDialog
137                 (
138                     {
139                         title:      "",
140                         content:    "",
141                         style:      "width: 95%;",
142                         closable:   true,
143                         onHide:    function ()
144                         {
145                             _this.fSemaphore.resolve ();
146                         }
147                     }
148                 );
149                 this.fDialog.startup ();
150 
151                 this.fNodeHeading = domConstruct.create 
152                 (
153                     "h1",
154                     {
155                     },
156                     this.fDialog.containerNode,
157                     "only"
158                 );
159                 this.fNodeText = domConstruct.create
160                 (
161                     "textarea",
162                     {
163                         style:      "width:100%;height:460px;font-family:monospace;resize:none",
164                         readonly:   true
165                     },
166                     this.fDialog.containerNode,
167                     "last"
168                 );
169             },
170         };
171     
172         ret = declare ("TTextWindow", [_WidgetBase], TTextWindow);
173     
174         return ret;
175     }
176 );
