 42         /* Removes leading spaces and deletes the last empty line 
 43          * from each node in the given node list.                               [10] */
 44         var NormalizeSourceListings = function ()
 45         {
 46             var kQuerySection       = "pre.prettyprint";
 47             
 48             var i;
 49             var n;
 50             var nL;
 51             var replPattern;
 52             var replReLeadSp;
 53             var replReSp;
 54             var nSpaces;
 55             var isList;
 56             var match;
 57             var doOmitLastLine;
 58             var sourceSections;
 59             var codeListing;
 60             var codeLine;
 61             var leadingSpaces;
 62             var newCode;
 63             
 64             sourceSections = query (kQuerySection);
 65             sourceSections.forEach
 66             (
 67                 function (node)
 68                 {   
 69                     newCode     = "";
 70                     codeListing = node.innerHTML.split ("\n");
 71                     isList      = Array.isArray (codeListing);
 72                     if (isList)
 73                     {
 74                         n = codeListing.length;
 75                         /* Ascertain: Is last line empty? */
 76                         if (n >= 1)
 77                         {
 78                             codeLine        = codeListing [n-1];
 79                             match           = codeLine.match (/^\s*$/);
 80                             doOmitLastLine  = (match !== null);
 81                             if (doOmitLastLine)
 82                             {
 83                                 n--;
 84                             }
 85                         }
 86                         
 87                         /* Delete leading spaces, but preserve indentation */
 88                         if (n >= 1)
 89                         {
 90                             codeLine = codeListing [0];
 91                             match    = codeLine.match (/^\s+/);
 92                             if (match !== null)
 93                             {
 94                                 leadingSpaces = match [0];
 95                                 nSpaces       = leadingSpaces.length;
 96                                 if (nSpaces >= 1)
 97                                 {
 98                                     replPattern     = "^\\s{" + nSpaces + "}";
 99                                     replReLeadSp    = new RegExp (replPattern);
100                                     replReSp        = new RegExp ("\\s", "g");  /* [20] */
101                                     for (i = 0; i < n; i++)
102                                     {
103                                         codeLine = codeListing [i];
104                                         codeLine = codeLine.replace (replReLeadSp,  ""          );
105                                         codeLine = codeLine.replace (replReSp,      " "    );
106                                         nL       = (i <= n-2)  ?  "\n"  :  "";
107                                         newCode += codeLine + nL;
108                                     }
109                                     
110                                     node.innerHTML = newCode;
111                                 }
112                             }
113                         }
114                     }
115                 }
116             );
117         };

221 /*
222     [1]  The decl script provides basic declarations for all the example scripts 
223          we load. We can't put those declarations into the callback function
224          inside the define directive, but must include the decl script before 
225          we load any of the example scripts. If we put the declarations into the 
226          callback, they will execute too late with the result that the example 
227          scripts will fail.
228 
229     [2]  We have to use a functor, otherwise we get a nullpointer error when clicking
230          a menu item.
231 
232     [10] We have to remove the leading space at each line of source code (code 
233          examples) without destroying the indentation. Basically, in each 
234          source code section we observe the leading spaces of the first line and
235          then remove those number of leading from all the lines in the section.
236          This follows the assumption that in all source code examples the first 
237          line is in indentation level zero (i.e. first line is the pilot line).
238 
239          This answers the problem that the css rule "white-space: pre-line;" 
240          is insufficient to clear all prepended spaces. As one solution,we 
241          tried to not have the leading spaces in the first place , but this 
242          confuses the indentation of the HTML code, making the HTML code harder 
243          to read.
244 
245          For each node:
246              1. Get innerHTML (content)
247              2. split content by newlines -> into Array
248              3. Isolate prepended spaces on first line
249              4. For each element in array, replace prepended spaces by empty string
250              5. For each element in array, replace space by non-breaking space
251              6. assemble new string from array (concat all elements, separated by newlines).
252              7.     but omit the last line if it is empty (0 or more spaces only).
253              8. set innerHTML to new string.
254 
255     [20] the present version of Chrome doesn't support flags in the String::replace()
256          function. But the 'global' flag works when specified inside a regular expression.
257          We have to use the 'global' flag, otherwise the String::replace() function 
258          will only replace the first occurence of a space.
259 
260  */
