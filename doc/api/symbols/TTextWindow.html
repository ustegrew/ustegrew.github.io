Class Detail
TTextWindow()
A text box with heading and copyable text area below. Used for the "Export to clipboard" feature.
The text window can be used with a callback (when the window is closed by the user) or via the Promise API (Promise resolves with closing the window).


Elements on the user interface 
.--------------------------------------------------------------.
| Title                                                        |
|--------------------------------------------------------------|
| Heading                                                      |
|                                                              |
| ************************************************************ |
| * Text                                                     * |
| *                                                          * |
| ************************************************************ |
'--------------------------------------------------------------'
Title
The dialog's title
Heading
The heading above the text.
Text
The text. Will be inside a TEXTAREA element so it's easy to copy to clipboard.


Example uses 
// Using callbacks

require 
(
    [
        "courseware/gui/TTextWindow/TTextWindow"
    ],
    function 
    (
        TTxtWnd
    )
    {
        var t;
        
        t = new TTxtWnd
        (
            host:        window,
            onClose: function ()
            {
                window.alert ("Bye!");
            }
        );
        t.startup ();
        
        t.Show
        (
            "Message to you",
            "This is a heading", 
            "And this is a text below the heading"
        );
    }
);
// Using promise API

require 
(
    [
        "courseware/gui/TTextWindow/TTextWindow"
    ],
    function 
    (
        TTxtWnd
    )
    {
        var t;
        
        t = new TTxtWnd
        (
            host:        window,
            onClose:     function () {}
        );
        t.startup ();
        
        t.Show
        (
            "Message to you",
            "This is a heading", 
            "And this is a text below the heading"
        ).then
        {
            function ()
            {
                window.alert ("Bye!");
            }
        };
    }
);
