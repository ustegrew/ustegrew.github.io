Class Detail
TButtonDialog()
A dialog showing a text and a set of buttons. Useful for multiple choice questions other than "About to reconfigure X... (OK? Cancel?)".
The dialog can be used with callbacks (one for each button's click event) or via the Promise API (Promise resolves with the index of the button clicked).


Elements on the user interface 
.----------------------------------------------------------------.
| Title                                                          |
|----------------------------------------------------------------|
| Question                                                       |
|                .-----------. .-----------.     .-------------. |
|                | Button[0] | | Button[1] | ... | Button[n-1] | |
|                '-----------' '-----------'     '-------------' |
'----------------------------------------------------------------'
Title
The dialog's title
Question
The question on which the user has to decide.
Button [0] ... Button [n-1]
The buttons representing the user's answer.


Example uses 
// Using callbacks

require 
(
    [
        "courseware/gui/TButtonDialog/TButtonDialog"
    ],
    function 
    (
        TBtnDlg
    )
    {
        var dlg;
        
        dlg = new TBtnDlg
        (
            host:        window,
            buttons:
            [
                {
                    label:       "Tell me what's 1 + 1!",
                    onClick: function ()
                    {
                        window.alert ("2");
                    }
                },
                {
                    label:       "I used to be decisive...",
                    onClick: function ()
                    {
                        window.alert ("... but now I'm not so sure.");
                    }
                },
            ]
        );
        dlg.startup ();
        
        dlg.Show
        (
            "Important things box", 
            "Choose today's important thing to contemplate on"
        );
    }
);
// Using promise API

require 
(
    [
        "courseware/gui/TButtonDialog/TButtonDialog"
    ],
    function 
    (
        TBtnDlg
    )
    {
        var dlg;
        
        dlg = new TBtnDlg
        (
            host:        window,
            buttons:
            [
                {
                    label:       "Tell me what's 1 + 1!",
                    onClick: function () {}
                },
                {
                    label:       "I used to be decisive...",
                    onClick: function () {}
                },
            ]
        );
        dlg.startup ();

        dlg.Show 
        (
            "Important things box", 
            "Choose today's important thing to contemplate on"
        )
        .then 
        {
            function (decision)
            {
                switch (decision)
                {
                    case 0:
                        window.alert ("2");
                        break;
                    case 1:
                        window.alert ("... but now I'm not so sure.");
                        break;
                    default:
                        window.alert ("Unknown button alert!);
                }
            }
        };
    }
);
